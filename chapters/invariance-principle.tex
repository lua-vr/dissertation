\section{Invariance Principle}

\begin{definition}
  A partially hyperbolic diffeomorphism is \textit{accessible} if between any two points in the manifold, there is a path composed of segments tangent to either \(E^s\) or \(E^u\).
\end{definition}

\begin{definition}
  A partially hyperbolic diffeomorphism is \textit{dynamically coherent} if there exists invariant foliations \(\fF^{cs}\) and \(\fF^{cu}\) tangent to \(E^c \oplus E^s\) and \(E^c \oplus E^u\), respectively.
\end{definition}

\begin{theorem}\label{thm:invariance-principle-skew-products}
  Let \(f \colon M \to M\), where \(M = X \times Y\), be a skew product diffeomorphism with an ergodic invariant measure \(\mu\).  Then, \(\lambda^c(\mu) \le 0\) implies that \(\set{\mu^c_x}_{x \in M}\) is invariant by unstable holonomies, meaning that \((h^u_{x,y})_*\mu_x^c = \mu^c_{y}\).
\end{theorem}
