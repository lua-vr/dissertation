\section{Probability spaces}

\subsection{Products}
A fact that is not always covered in measure theory courses and literature is that we can take arbitrary products of measures when they are probability measures. We will sneakily use this fact in some definitions of this section.

\begin{theorem}
  Let \(T\) be an arbitrary index set, and let \((X_t,\mathcal{B}_t,\mu_t)\) be a probability space. Then, there exists a unique measure \(\mu\) on the measurable space \(\rpar{\prod_{t \in T}X_t, \bigotimes_{t \in T}\mathcal{B}_t}\), which is the product of \(\mu_t\). That is, for all finite subsets \(\Lambda \subseteq T\), considering the canonical projection \(\pi_{\Lambda}\colon \prod_{t \in T}X_t \to \prod_{t \in \Lambda}X_t\), we have \((\pi_{\Lambda})_*\mu = \bigotimes_{t \in \Lambda}\mu_t\).
\end{theorem}

\begin{proof}
  See~\cite[section 3.5]{bogachevMeasureTheory2007}.
\end{proof}

\subsection{The Kolmogorov Extension Theorem}

Let \(T\) be an arbitrary index set. Suppose that for every \(t \in T\), we have a measurable space \((\Omega_t, \mathcal{B}_t)\). Moreover, suppose we have a more complex specification of how to ``patch together'' finite collections of these measurable spaces into a measure spaces, that is itself consistent: for every finite subset \(\Lambda \subset T\), we specify a measure \(\mu_{\Lambda}\) on \((\Omega_{\Lambda} \coloneqq \prod_{t \in \Lambda}\Omega_t
, \mathcal{B}_{\Lambda} \colon= \bigotimes_{t \in \Lambda} \mathcal{B}_t)\), such that if \(\Lambda_1 \subset \Lambda_2\), the image of the measure \(\mu_{\Lambda_2}\) under the natural projection from \(\Omega_{\Lambda_2}\) to \(\Omega_{\Lambda_1}\) coincides with \(\mu_{\Lambda_1}\). We call a such a specification \(\set{\mu_{\Lambda}}_{\Lambda}\) a \emph{projective system} of measures.

From this finitary specification, when is it possible to construct a measure on the whole product $\left(\Omega:=\prod_{t \in T} \Omega_t, \mathcal{B}:=\bigotimes_{t \in T} \mathcal{B}_t\right)$? Clearly, this is a more general problem than that of constructing an infinite product measure. In fact, this is not always possible in general, but as we shall see, a topological-flavored condition proves to be sufficient for the existence of the extension.

\begin{definition}
  Let \(X\) be a set. A collection \(\mathcal{K} \subseteq 2^X\) of subsets is called a \emph{compact class} if it has the finite intersection property, meaning that for all countable subsets \((K_n)_{n = 1}^{\infty} \subseteq \mathcal{K}\), then
  \[\rpar[\Big]{\forall N,\;\; \bigcap_{n = 1}^N K_n \neq \varnothing} \implies \bigcap_{n = 1}^{\infty}K_n \neq \varnothing.\]
\end{definition}

\begin{definition}
  Let \((X,\mathcal{B}_X,\mu)\) be a measure space. A compact class \(\mathcal{K} \subseteq \mathcal{B}_X\) is called \emph{approximating} if the values of \(\mu\) on any set can be approximated by those on elements of \(\mathcal{K}\), meaning that for any measurable \(A\), and \(\varepsilon > 0\), there exists \(K \in \mathcal{K}\) such that \(K \subseteq A\) and \(\mu(A \setminus K) < \varepsilon\).
\end{definition}

\begin{theorem}[Kolmogorov Extension Theorem]\label{thm:kolmogorov-ext}
  Suppose that for every $t \in T$, the measure $\mu_{\set{t}}$ on $\mathcal{B}_{\set{t}}$ possesses an approximating compact class $\mathcal{K}_{t} \subset \mathcal{B}_{\set{t}}$. Then, there exists a probability measure $\mu$ on the measurable space $(\Omega, \mathcal{B})$ such that the image of $\mu$ under the natural projection from $\Omega$ to $\Omega_{\Lambda}$ is $\mu_{\Lambda}$ for each finite set $\Lambda \subset T$.
\end{theorem}

\begin{proof}
  See~\cite[theorem 7.7.1]{bogachevMeasureTheory2007}.
\end{proof}

\begin{remark}
  Clearly, if each \(\mathcal{B}_{\set{t}}\) is the Borel algebra of a topological space whose compact sets satisfy the finite intersection property and \(\mu_{\set{t}}\) is inner regular, then it has an approximating compact class given by the compact sets. For instance, this is the case in any Standard Borel space, a family which includes all finite Borel measures in complete separable metric spaces, like \(\RR\).
\end{remark}

\section{Random variables}

If \((X, \mathcal{A}, \Prob)\) is a probability space and \((Y, \mathcal{B})\) is a measurable space, then a \emph{random variable} is simply a measurable function \(f \colon X \to Y\). It can be interpreted as some observable quantity of the probability space. If \(\mu\) is a measure on \(Y\), we may write \(f \sim \mu\) to mean that \(f_*\Prob = \mu\), and we say that \(f\) has distribution \(\mu\).

\begin{definition}
  We say the collection \(\xi = (\xi_i)_{i \in \mathcal{I}} \subseteq X^Y\) of random variables is \textit{independent} if \(\xi_*\Prob\) is a product measure, that is, if
  \begin{equation*}
    \xi_*\Prob = \bigotimes_{i \in \mathcal{I}}(\xi_i)_*\Prob.
  \end{equation*}
  This is equivalent the usual notion that \(\xi\) is independent if for every finite subset \(J \subseteq \mathcal{I}\) and for every collection of measurable sets \(A_j \in \mathcal{B}\), we have
  \begin{equation*}
    \Prob \rpar[\bigg]{\bigcap_{j \in J} \xi_j^{-1}(A_j)} = \prod_{j \in J} \Prob \rpar*{\xi_j^{-1}(A_j)}.
  \end{equation*}
\end{definition}

\section{Weak convergence}

\begin{proposition}[Portmanteau]\label{thm:portmanteau}
  The following are equivalent:
  \begin{enumerate}[i., left=0pt, topsep=3pt]
    \item \(\mu_n \weakto \mu\);
    \item \(\lim_{n \to \infty} \mu_n(\phi) = \mu(\phi)\) for all Lipschitz functions \(\phi\);
    \item \(\liminf_{n \to \infty} \mu_n(U) \geq \mu(U)\) for all open sets \(U\);
    \item \(\limsup_{n \to \infty} \mu_n(F) \leq \mu(F)\) for all closed sets \(F\);
    \item \(\lim_{n \to \infty} \mu_n(A) = \mu(A)\) for all continuity sets \(A\) of the measure \(\mu\).
  \end{enumerate}
\end{proposition}

\section{Brownian motion}\label{sec:brownian-motion}

The rigorous mathematical foundation of Brownian motion was laid by Norbert Wiener, who constructed Brownian motion as a measure on the space of continuous functions. This construction is why the Brownian motion is often called the Wiener process, and defined as the \emph{Wiener measure}.

We shall denote by \(\Normal[\mu][\nu] \in \Meas[1](\RR)\) the probability measure corresponding to the normal distribution with mean \(\mu\) and variance \(\nu\). It can be defined as \(\Normal[\mu][\nu] = \rho \Leb\), where \(\rho\) is the probability density function
\begin{equation*}
  \rho(x) = \frac{1}{\sqrt{2\pi \nu}}\exp\rpar*{-\frac{(x - \mu)^2}{2\nu}}.
\end{equation*}

\begin{definition}
  Let \(\mathcal{C}\) be the product \(\sigma\)-algebra on \(\CWPS\). The Wiener measure \(\WM\) is the unique measure on \((\CWPS, \mathcal{C})\) defined by the following properties: if we denote by \(\WM[t] \colon \CWPS \to \RR\) the evaluation at time \(t\), seen as a random variable on \((\CWPS, \mathcal{C}, \WM)\), then
  \begin{enumerate}[(i)]
  \item \(\WM[0] = 0\) almost surely;
  \item for \(t > 0\), \(\WM[t] \sim \Normal[0][t]\);
  \item if \(0 \leq t_1 < t_2 < \dots < t_n\), then the increments \(\WM[t_1], \WM[t_2] - \WM[t_1], \dots, \WM[t_n] - \WM[t_{n - 1}]\) are independent random variables.
  \end{enumerate}
\end{definition}

\subsection{The space of càdlàg functions}\label{sec:space-cadl-funct}

The space \(\CWPS\) of continuous functions does not let us talk about processes that contain discontinuities, or even about how discontinuous processes can converge in a suitable way to continuous ones. This is important in our setting to deal with the convergence of certain processes induced by discrete dynamics, which consist of step functions, to the Brownian motion, which is continuous almost surely.

The space of \textit{càdlàg} functions, or Skorohod space, is the space of functions whose
discontinuities are of the \textit{first kind}, that is, both lateral limits exist, and whose value
at the point is equal to the right limit:
\begin{equation*}
  \Sko{[0,1]} = \set[par=\Big]{ f \colon [0,1] \to \RR \where \forall t, \lim_{s \to t^-}x(s) \text{ exists and} \lim_{s \to t^+}x(s) = x(t) }.
\end{equation*}
We want to put a metric on this space with the property that functions are close together if their values are close to each other uniformly, but also allowing small perturbations in the parametrization of time. 

\nomenclature{\(\Cs{X}\)}{The set of continuous functions \(X \to \RR\) with the topology of uniform convergence.}
\nomenclature{\(\Sko{[0,1]}\)}{The Skorohod space, defined in \cref{sec:space-cadl-funct}.}

\begin{definition}[Skorohod topology]
  Let \(\Lambda\) be the set of strictly increasing continuous functions from \([0,1]\) to itself. For \(f,g \in \WPS\), we define
  \begin{equation}\label{eq:sko-metric}
    d(f,g) = \inf_{\lambda \in \Lambda} \max\set[par = \big]{ \lVert \lambda - \Id \rVert_{\infty}, \lVert f - g \circ \lambda \rVert_{\infty}}.
  \end{equation}
\end{definition}
The distance above defines a metric on \(\WPS\), whose topology is called the Skorohod topology. Since \(d(f,g) \leq \lVert f - g \rVert_{\infty}\), the Skorohod topology is coarser than the uniform topology.

\begin{proposition}\label{prop:skorohod-uniform}
  The restriction of the Skorohod topology to \(\CWPS\) coincides with the uniform topology. In fact, if \(f \colon [0,1] \to \RR\) is continuous, then any ball \[B_{\delta}(f) = \set{g \in \Sko{[0,1]} \where \norm{f - g}_{\infty} < \delta},\] of the uniform topology contains an open neighborhood of \(f\) in the Skorohod topology.
  % That is, there exists \(\delta'\) such that
  % \begin{equation*}
  %   B_{\delta'}(f) \subseteq B_{\delta'}'(f) \subseteq  B_{\delta}(f)
  % \end{equation*}
  % where \(B_{\delta'}'(f) = \set{g \in \Sko{[0,1]} \where d(f, g) < \delta}\).
\end{proposition}
\begin{proof}
  Since the Skorohod topology is coarser, it suffices to show the second part of the statement. Since \(f\) is uniformly continuous, there is \(\delta_2 < \delta/2\) such that if \(\norm{\lambda - \Id}_{\infty} < \delta_2\), then
  \begin{equation*}
    \norm{f - f \circ \lambda}_{\infty} < \delta/2.
  \end{equation*}
  Let \[B'(f) = \set{g \in \Sko{[0,1]} \where d(g,f) < \delta_2/2}.\] The set \(B'(f)\) is a ball of the Skorohod topology, so it suffices to show \(B'(f) \subseteq B_{\delta}(f)\). Let \(g \in B'(f)\). Since \(\delta_2/2 < \delta_2\), there exists \(\lambda\) such that 
  \begin{equation*}
    \max\set[par = \big]{ \lVert \lambda - \Id \rVert_{\infty}, \lVert g - f \circ \lambda \rVert_{\infty}} < \delta_2 < \delta/2.
  \end{equation*}
  Therefore,
  \begin{equation*}
    \lVert g - f \rVert_{\infty} \leq \lVert g - f \circ \lambda \rVert_{\infty} +  \lVert f \circ \lambda - f \rVert_{\infty} < \delta.
  \end{equation*}
\end{proof}

\begin{proposition}\label{prop:sko-conv-pointwise}
  If \(f_n \to f\) in the Skorohod topology, then \(f_n(x) \to f(x)\) for all \(x\) in a set \(E\) whose complement is at most countable.
\end{proposition}

\begin{proof}
  Since \(f \in \WPS\), the set of discontinuities of \(f\) is at most countable\footnote{Proof: let \(t\) be a discontinuity point of \(f\), and let \(U\) be a neighborhood of \(t\) where the value of the function is close to the lateral limits at \(t\). Then, \(f\) is continuous at \(U \setminus \set{t}\), which is open, and therefore contains a rational point \(q_t\). With some care, this will define an injection from discontinuity points to rationals.}. Let \(E\) be a cocountable set such that \(f|_E\) is continuous, and let \(x \in E\) and \(\varepsilon > 0\). Pick \(\delta\) so that \(\abs{x - y} < \delta\) implies \(\abs{f(x) - f(y)} < \varepsilon\), and let \(N\) be big enough so that \(n \ge N\) implies \(d(f,f_n) < \min\set{\delta, \varepsilon}/2\). If \(n \ge N\), and if we pick a \(\lambda \in \Lambda\) that makes the right side of \cref{eq:sko-metric} \(\min\set{\delta, \varepsilon}/2\)-close to the infimum, then
  \begin{align*}
    \abs{f(x) - f_n(x)} &\le \abs{f(x) - f \circ \lambda ^{-1}(x)} + \abs{f \circ \lambda ^{-1}(x) - f_n(x)}\\
                        &\le  \varepsilon + \norm{f - f_n \circ \lambda}_{\infty} \le 2\varepsilon,
  \end{align*}
  because \(\abs{\lambda ^{-1}(x) - x} \le \norm{\lambda - \Id}_{\infty} < \delta\).

  
\end{proof}

\subsection{The arcsine law}

\begin{definition}
  We shall denote by \(\pTime \colon \WPS \to \RR\) the observable that returns the proportion of time its function argument is positive, that is,
  \[\pTime{f} = \Leb \{ x \in [0,1] : f(x) > 0 \} = \int_{[0,1]} 1_{(0,\infty)} \circ f\;d\Leb.\]
\end{definition}


\begin{lemma}\label{lem:ptime-continuous}
  \(\pTime\) is measurable and continuous in a full \(W\)-measure set.
\end{lemma}

\begin{proof}
  Since the normal distributions are non-atomic and \(W_t\) has normal distribution for every \(t \in \interval[open left]{0}{1}\), we have that \[W(\{ f \in \WPS : f(t) = 0 \}) = 0\] for all \(t \in \interval[open left]{0}{1}\). Therefore, by Fubini, \[\Leb(\{ t \in [0,1] : f(t) = 0 \}) = 0\] for all \(f\) in a set \(E \subseteq \WPS\) of full \(W\)-measure. It remains to check that \(\pTime\) is continuous at such points \(f \in E\). Indeed, \(0\) is the only discontinuity of the indicator function \(1_{(0,\infty)}\), so if \(f \in E\) and \(f_n \to f\) in the Skorohod space, then \(1_{(0,\infty)} \circ f_n \to 1_{(0,\infty)} \circ f\) almost everywhere by \cref{prop:sko-conv-pointwise}. By the dominated convergence theorem,
  \begin{equation*}
    \pTime{f} =  \int_{[0,1]} 1_{(0,\infty)} \circ f \;d\Leb = \lim_{n \to \infty}
    \int_{[0,1]} 1_{(0,\infty)} \circ f_n \;d\Leb = \lim_{n \to \infty} \pTime{f_n}.
  \end{equation*}
\end{proof}

\begin{theorem}[Arcsine law]\label{thm:arcsine-brownian}
  For \(\alpha \in [0,1]\),
  \[\pTime[spar,pf]W([0,\alpha]) = \frac{2}{\pi} \arcsin \sqrt{\alpha}.\]
\end{theorem}

\begin{proof}
  See~\cite[equation 9.27]{billingsleyConvergenceProbabilityMeasures1999}.
\end{proof}

\begin{theorem}[Forgery Theorem, or Support Theorem]\label{thm:forgery}
  If \(f \in \CWPS\) and \(f(0) = 0\), then every neighborhood of \(f\) has positive Wiener measure. Note that by \cref{prop:skorohod-uniform}, this is true for neighborhoods both in the uniform and in the Skorohod topologies.
\end{theorem}

\begin{proof}
  See, for instance,~\cite{peresAnswerWienerMeasure2022}. This is known as the Stroock-Varadhan Support Theorem or Lévy's Forgery Theorem.
\end{proof}

\subsection{Weak invariance principles}

We will fix a probability space \((\Omega, \mathcal{F}, \Prob)\) for the rest of this section. Given a sequence \(\xi = (\xi_0,\xi_1,\dots)\) of random variables defined on \(\Omega\), we can define their partial sums \(\PSum[n,rv=\xi] = \sum_{i = 0}^{n - 1}\xi_i\) and a sequence of processes \(\DProc[n,rv=\xi] \colon \Omega \to \WPS\) given by
\[\DProc[n,rv=\xi]{\omega}{t} = \frac{\PSum[\floor{nt},rv=\xi]{\omega}}{\sqrt{n}},\]
obtained by appropriately scaling the partial sums of the random variables up to some iterate \(n\).

Remarkably, for many classes of sequences \(\xi\), their corresponding processes \(\DProc[n]\) converge in some way to the Brownian motion --- we denote by an \emph{invariance principle} a result of this form. Perhaps the best known invariance principle is the Donsker's theorem, which is a \emph{weak} invariance principle since it asserts convergence in the weak topology.

\begin{theorem}[Donsker]\label{thm:donsker}
  If \(\xi \subseteq L^2(\Prob)\) is a sequence of \iid variables with mean zero and variance \(1\), then the measures \(\DProc[n,spar,pf] \Prob\) converge weakly to the Wiener measure. In other words, for every measurable continuous function \(\varphi \colon \WPS \to \mathbb{R}\),
  \[\int \varphi \circ \DProc[n] \;d\Prob \to \int \varphi \;dW.\]
\end{theorem}

The motivation for looking into invariance principles is that they allow transferring certain results from Brownian motions to the sequence of partial sums \(S^{\xi}_n\). In our case, the result we are interested in is the arcsine law. We know from \cref{lem:ptime-continuous} that \(\pTime\) is continuous \almostevery, so weak convergence is enough to transfer its conclusions:

\begin{lemma}\label{lem:weak-arcsine}
  Suppose \(\xi \subseteq L^2(\Prob)\) is such that \(\DProc[n,spar,pf]\Prob \weakto W\). Then the average of the time in which the random walk \(S_n^{\xi}\) is above \(M \in \RR\) converges to an arcsine distribution, in the sense that
  \begin{equation*}
    \lim_{n \to \infty}\Prob[par=auto]{\set[par=auto]{\omega \in \Omega \where \frac{1}{n} \#\set{ 1 \leq i \leq n \where S^{\xi}_n(\omega) > M } \leq \alpha}} = \frac{2}{\pi} \arcsin \sqrt{\alpha}.
  \end{equation*}
\end{lemma}

\begin{proof}
  Writing \(\pTime[n]{\omega} = \frac{1}{n} \#\set{ 1 \leq i \leq n \where S^{\xi}_n(\omega) > M }\), the left-hand side can be rewritten as \(\lim_{n \to \infty}\pTime[n,spar,pf]\Prob{[0,\alpha]}\). Set \(c_n = M/\sqrt{n}\), and notice that
  \[\pTime[n]{\omega} =\frac{1}{n} \#\set[par = auto]{ 1 \leq i \leq n \where \frac{S^{\xi}_i(\omega)}{\sqrt{n}} - \frac{M}{\sqrt{n}} >0 } = \pTime{\DProc[n,rv = \xi]{\omega} - c_n}.\]
  Since by \cref{thm:arcsine-brownian} we know \(Z = \pTime[spar,pf]W\) is equivalent to the Lebesgue measure on \([0,1]\), intervals are continuity sets for it. By the weak convergence assumption and continuity of \(\pTime\), we know \(\Fun{\pTime \circ \DProc[n,rv = \xi]}[spar,pf]\Prob\weakto Z\), and since \(c_n \to 0\) (as a sequence in \(\WPS\)), in fact \[\Fun{\pTime \circ (\DProc[n,rv = \xi] - c_n)}[spar,pf]\Prob\weakto Z.\] By Portmanteau, this means 
  \[\lim_{n \to \infty}\pTime[n,spar,pf]\Prob{[0,\alpha]} = \lim_{n \to \infty}\Fun{\pTime \circ (\DProc[n,rv = \xi] - c_n)}[spar,pf]\Prob{[0,\alpha]} = \frac{2}{\pi}\arcsin \sqrt{\alpha}.\]
\end{proof}

\subsubsection{Ergodic Martingale Differences}\label{sec:ergod-mart-diff}

Donsker's theorem is not the most general result of its kind. In our case, we are interested in a version of the weak invariance principle for ergodic martingale difference sequences. Intuitively, these are the sequences that behave as the time-one steps of a martingale, in the sense that even if the value of some previous steps are known, the expected value of the next step is still zero.

\begin{definition}
  A sequence \(\xi = (\xi_0,\xi_1\dots) \in L^1(\Prob)^{\NN}\) of random variables  is said to be a \emph{martingale difference sequence} if for all \(n\),
  \begin{equation*}
    \CE[\Prob]{\xi_{n}, {\sigma(\xi_0,\dots,\xi_{n - 1})}} = 0.
  \end{equation*}
\end{definition}

\subsubsection{Stationarity and two-sided processes}

In the previous definition we considered an one-sided sequence of random variables. As is often the case in ergodic theory, the sequences we shall be dealing with have the additional property of \emph{stationarity}, which is a sort of time-invariance. This property will also allow us to lift one-sided sequences of variables to two-sided ones, that is, sequences with indices in \(\ZZ\), due to the existence of the invertible extensions exposed in~\cref{sec:invert-extens}.

\begin{definition}
  A sequence \(\xi = (\xi_0,\xi_1\dots)\) of random variables taking values in \(Y\) is said to be \emph{stationary} if \(\xi_*\Prob\) is an invariant measure for the left shift \(\sigma \colon Y^{\NN} \to Y^{\NN}\). Similarly, we can define stationarity for two-sided sequences by replacing \(\NN\) by \(\ZZ\) in the definition.
\end{definition}

\begin{definition}
  A two-sided sequence \(\xi \in L^1(\Prob)^{\ZZ}\) of random variables  is said to be a \emph{martingale difference sequence} if for all \(n\),
  \begin{equation*}
    \CE[\Prob]{\xi_{n}, {\sigma(\set{\xi_i \where i < n})}} = 0.
  \end{equation*}
\end{definition}

\begin{lemma}[Lévy's upward theorem]\label{lem:levy-up}
  Let \((X,\mathcal{B}\mu)\) be a measure space and \(\varphi \colon X \to \RR\) be a random variable. If \(\set{\mathcal{B}_n}_{n \in \NN}\) is an increasing sequence of sub-\(\sigma\)-algebras of \(\mathcal{B}\) (an upwards filtration), then
  \begin{equation*}
    \lim_{n \to \infty} \CE[\mu]{\varphi, \mathcal{B}_n} = \CE[\mu,par = \big]{\varphi, \sigma(\set{\mathcal{B}_n}_{n \in \NN})}
  \end{equation*}
  almost-surely.
\end{lemma}

\begin{proof}
  See~\cite{billingsleyProbabilityMeasure1995}.
\end{proof}

\begin{lemma}\label{lem:two-sided-from-one}
  Every stationary one-sided ergodic martingale difference sequence \((\xi, \Prob_1)\) can be extended to a two-sided stationary ergodic martingale difference sequence \((\zeta, \Prob_2)\) with the same marginal distributions on positive indices. That is, if \(\pi \colon \RR^{\ZZ}\to \RR^{\NN}\) is the canonical projection, we have \((\pi \circ \zeta)_*\Prob_2 = \xi_*\Prob_1\).
\end{lemma}

\begin{proof}
  Consider the projective system of measures \(\set{\mu_{\Lambda}}_{\Lambda}\) on \(\RR^{\ZZ}\) defined by
  \begin{equation*}
    \mu_{\set{z_1,z_2,\dots,z_n}} = (\pi_{\set{0, (z_2 - z_1),\dots,(z_n - z_1)}} \circ \xi)_*\Prob_1,
  \end{equation*}
  where for \(\Lambda \subseteq \NN\), \(\pi_{\Lambda} \colon \RR^{\NN}\to \RR^{\Lambda}\) is the canonical projection. Since \(\xi_*\Prob_1\) is invariant by the left shift \(\sigma\), this system satisfies the consistency condition of \cref{thm:kolmogorov-ext}. Let \(\Prob_2 \in \Meas{\RR^{\ZZ}}\) be the extension of the measure given by the theorem, and define \(\zeta_n \colon \RR^{\ZZ} \to \RR\) simply as \(\zeta_n((x_i)_i) = x_n\). By definition, it satisfies \((\pi \circ \zeta)_*\Prob_2 = \xi_*\Prob_1\).

  Now, we show it is a two-sided martingale difference sequence. By stationarity and the extension property, this two-sided extension satisfies
  \begin{equation*}
    \CE[\Prob_2]{\zeta_{n}, {\sigma(\set{\zeta_{n - k},\dots,\zeta_{n - 1}})}} = 
    \CE[\Prob_1]{\xi_{n + k}, {\sigma(\set{\xi_{0},\dots,\xi_{n - 1 + k}})}} = 0
  \end{equation*}
  for all \(k\). By letting \(k \to \infty\) and applying \cref{lem:levy-up}, this implies \(\CE[\Prob_2]{\zeta_{n}, {\sigma(\set{\zeta_i \where i < n})}} = 0\).
\end{proof}


\subsubsection{Invariance principle for ergodic sequences}

\begin{definition}
  A sequence \(\xi = (\xi_0,\xi_1\dots)\) of random variables is said to be \emph{ergodic} if \(\xi_*\Prob\) is an ergodic measure for the left shift \(\sigma \colon Y^{\NN} \to Y^{\NN}\). 
\end{definition}

\begin{example}
  Let \((\cms, f, \mu)\) be a measure-preserving system with an ergodic measure \(\mu\). Any function \(\varphi \colon X \to Y\) defines a sequence of random variables \[\xi = (\varphi \circ f^{n})_{n \ge 0},\]
  and this sequence can also be seen as a factor map for the extension \(\xi \colon (X,f,\mu) \to (Y^{\NN}, \sigma, \xi_*\mu)\). By applying \cref{lem:factor-ergodic}, we see that the sequence \(\xi\) is an ergodic sequence.
\end{example}

\begin{theorem}\label{thm:donsker-martingale}
  If \(\xi \subseteq L^2(\Prob)\) is a stationary ergodic martingale difference sequence for which \(\sigma = \norm{\xi_1}_2\) satisfies \(0 < \sigma < \infty\), then the measures \(\DProc[n, spar, pf] \Prob \weakto \sigma W\).
\end{theorem}

\begin{proof}
  See~\cite[theorem 18.3]{billingsleyConvergenceProbabilityMeasures1999}.
\end{proof}

\subsection{Reverse martingale difference sequences}

In ergodic dynamics, the processes we usually care about are the values \(\xi = (\varphi \circ f^{n})_{n \geq 0}\) of some observable \(\varphi \in L^1(\mu)\) along orbits of \(f\). This process is unlikely be a martingale difference sequence unless \(\varphi \equiv 0\), because \(\varphi \circ f^n\) is measurable for the \(\sigma\)-algebra \(f^{-n}(\mathcal{F})\), which is non-increasing due to measurability of \(f\). In other words, we may have \(\CE[\mu]{\varphi \circ f^{n}, {\sigma(\varphi, \dots, \varphi \circ f^{n})}} = \varphi \circ f^{n} \neq 0\) in general. In the non-invertible case, the algebras \(f^{-n}(\mathcal{F})\) form a decreasing sequence in general. This suggests we should look at the processes in the opposite direction, and motivates the following definition.
\begin{definition}
  A sequence \(\xi = (\xi_1,\xi_2\dots)\) of random variables is said to be a \textit{reverse} martingale difference sequence if for all \(n\),
  \begin{equation*}
    \CE[\Prob]{\xi_{n}, {\sigma(\set{\xi_i \where i > n})}} = 0.
  \end{equation*}
\end{definition}

In the case of a stationary sequence (as is the case for \(\xi = (\varphi \circ f^{n})_{n \geq 0}\) and \(\Prob = \mu\) an invariant measure for \(f\)), results for martingale difference sequences can be stated dually for reverse martingale difference sequences. The following discussion expands on comments and text present in~\cite[Appendix M22 and Section 19]{billingsleyConvergenceProbabilityMeasures1999}.

\begin{corollary}\label{cor:donsker-martingale}
  If \(\xi \in L^2(\Prob)^{\NN}\) is a stationary ergodic reverse martingale difference sequence for which \(\sigma = \norm{\xi_1}_2 > 0\), then the measures \({(D^{\xi}_n)}_* \Prob \weakto \sigma W\).
\end{corollary}

\begin{proof}
  First, apply the extension theorem as in \cref{lem:two-sided-from-one} in order to lift \(\xi\) to a two-sided sequence \(\zeta \in L^2(\Prob_2)^{\ZZ}\) given by the coordinates of the probability space \((\RR^{\ZZ},\Prob_2)\). Then, invert the sequence: consider \(\zeta'_i = \zeta_{-i}\) and the probability \(\Prob_3 = \tau_*\Prob_2\), where  \(\tau((x_i)) = (x_{-i})\). Then \((\zeta,\Prob_3)\) is an ordinary two-sided martingale difference sequence, for which we can apply \cref{thm:donsker-martingale}. Now, the key is noticing that due to commutativity, \(\zeta'_0 + \cdots + \zeta'_n = \zeta_{0} + \cdots + \zeta_{-n}\) is equal to \(\zeta_{-n} + \cdots + \zeta_0\), which by stationarity  has the same distribution as \(\zeta_0 + \cdots + \zeta_n\). The process \(\DProc[n,rv = \zeta]\) only depends on such sums, therefore
  \begin{equation*}
    \DProc[n, rv = \zeta', spar, pf] \Prob_3 = (\DProc[n, rv = \zeta])_* \Prob_2 = (\DProc[n, rv = \xi])_* \Prob
  \end{equation*}
  for all \(n\), which implies \({(D^{\xi}_n)}_* \Prob \weakto \sigma W\).
\end{proof}
