\begin{frame}{Cross ratio and Schwarzian derivative}
  \begin{definition}[Cross ratio]
    \begin{equation*}
      \rho(x_0,x_1,x_2,x_3) = \frac{(x_2 - x_0)(x_3 - x_1)}{(x_1 - x_0)(x_3 - x_2)}
    \end{equation*}
  \end{definition}
  \begin{definition}
    Let \(\f \colon I \to I'\) be a \(\Cs[d = 3]\) diffeomorphism between two intervals of the real line.
    The \emph{Schwarzian derivative} of \(\f\) is the function
    \begin{equation*}
      \SchD\f = \frac{\f[der = 3]}{\f[der = 1]} - \frac{3}{2}\rpar*{\frac{\f[der = 2]}{\f[der = 1]}}^2.
    \end{equation*}
  \end{definition}
\end{frame}

\begin{frame}{Cross ratio and Schwarzian derivative}
  \begin{lemma}\label{prop:schwarzian}
    The Schwarzian derivative has the following properties:
    \begin{enumerate}
    \item\label{prop:schwarzian:item:1}
      If \(\mathcal{S}f \succ 0\) and \(\mathcal{S}g \succ 0\), then \(\mathcal{S}(f \circ g) \succ 0\), and similarly for \(\prec 0\). Moreover, if \(\mathcal{S}f \equiv 0\), then \(\SchD{f \circ g} = \SchD g \cdot (g')^2\), and if \(\SchD g \equiv 0\), then \(\SchD{f \circ g} = (\SchD f) \circ g\).
    \item If \(\mathcal{S}f \prec 0\), then \(\mathcal{S}f^{-1} \succ 0\). 
    \item\label{prop:schwarzian:item:3} \(\SchD f \prec 0\) if, and only if, \(\varphi(y) = 1/ \sqrt{\lvert f'(y) \rvert}\) is strictly convex.
    \item \(\SchD f \equiv 0\) if, and only if, \(f\) is a fractional linear transformation.
    \end{enumerate}
  \end{lemma}
\end{frame}

\begin{frame}
  \frametitle{Negative Schwarzian}
  \begin{theorem}\label{thm:neg-schwarz-sep-graph}
    If \((x,y) \mapsto \SchD f_x(y)\) is continuous and \(\SchD f_x \prec 0\) almost everywhere in \(x \in X\), then there exists a measurable function \(\sigma \colon X \to I\) defined almost everywhere such that
    \begin{equation*}
      (x,y) \in \Basin{\mu_0} \text{ whenever } y < \sigma(x) \quad \text{and} \quad (x,y) \in \Basin{\mu_1} \text{ whenever }y > \sigma(x).
    \end{equation*}
    Consequently, the union \(\Basin{\mu_0} \cup \Basin{\mu_1}\) has total \(\mu \otimes \Leb\) measure.
  \end{theorem}
\end{frame}

\begin{frame}
  \frametitle{Positive Schwarzian}
  \begin{theorem}\label{thm:pos-schwarz-asymptotic}
    If \(\SchD f_x \succ 0\) almost everywhere, then \(F\) has a unique asymptotic measure \(\nu\) for \(\mu \otimes \Leb\)-almost every point. Furthermore, for each \(i \in \{ 0,1 \}\), \(\Lyap(\mathcal{A}_i) > 0\) implies \(\nu(\mathcal{A}_i) = 0\).
  \end{theorem}
  \begin{proof}[Idea]
    Consider the inverse of the natural extension \(\tilde{F}^{-1}\). It satisfies \(\SchD f_{\tilde x}^{-1} \prec 0\) \(\tilde \mu\)-almost everywhere, and therefore has a separating graph \(\sigma\) that is repelling.\\[1em]

    The orbits for \(\tilde{F}\) will be attracted by the graph, and \(\tilde \nu = (\graph \sigma)_*\tilde \mu\) is asymptotic for \(\tilde{F}\). Pushing it to the base via \(\pi\), we have the asymptotic measure \(\nu = \pi_*\tilde \nu\).
  \end{proof}
\end{frame}

\section{Partially hyperbolic dynamics}

\begin{frame}
  \frametitle{Dominated splittings}
  \begin{definition}
    We say \(TM = E_1\oplus \cdots \oplus E_n\) is a \emph{splitting} of \(TM\) if at each \(x \in M\), \(T_xM = E_1(x)\oplus \cdots \oplus E_n(x)\) and \(\dim E_i(x)\) does not depend on \(x\).
  \end{definition}
  \begin{definition}
    We say a splitting \(TM = E_1\oplus \cdots \oplus E_n\) is \emph{dominated} for a diffeomorphism \(f \colon M \to M\) if:
    \begin{enumerate}
    \item the splitting is \(df\)-\emph{invariant}, that is, \(df_x(E_i(x)) = E_i(f(x))\) for all \(i,x\);
    \item there exists constants \(c > 0\) and \(\lambda \in (0,1)\) such that for all \(1 \le i \le n - 1\), \(x \in M\) \(u \in E_i(x) \setminus \set{0}\), \(v \in E_{i + 1}(x) \setminus \set{0}\) and \(k \ge 0\),
      \begin{equation*}
        \frac{\norm{df^k_x(u)}}{\norm{u}} \le c \lambda^k \frac{\norm{df^k_x(v)}}{\norm{v}}.
      \end{equation*}
    \end{enumerate}
  \end{definition}
\end{frame}

\begin{frame}
  \frametitle{Partial hyperbolicity}
  \begin{definition}
    A subbundle \(E \subseteq TM\) is \emph{uniformly contracted} by \(df\) if there exists \(c > 0\) and \(\lambda \in (0,1)\) such that for all \(x \in M\), \(u \in E(x)\) and \(k \ge 0\),
    \begin{equation*}
      \norm{df^k_x(u)} \le c \lambda^k \norm{u}.
    \end{equation*}
  \end{definition}

  \begin{definition}
    Let \(f \colon M \to M\) be a diffeomorphism. We say \(f\) is \emph{partially hyperbolic} if there exists a dominated splitting \(TM = E^s \oplus E^c \oplus E^u\) such that \(E^s\) is uniformly contracted by \(df\) and \(E^u\) is uniformly contracted by \(df ^{-1}\).
  \end{definition}
\end{frame}




\begin{frame}{Invariance principle}
\begin{definition}
  Consider the quotient map \(\pi \colon X \to X/\mathcal{F}^c\). We say that a partially hyperbolic diffeomorphism \(f \colon X \to X\) has \emph{global unstable holonomies} if for every \(p_1,p_2 \in X/\mathcal{F}^c\) that are in the same leaf of \(\pi(\mathcal{F}^{u})\), there exists an homeomorphism \(h^u_{p_1,p_2}\colon \mathcal{F}^c(p_1) \to \mathcal{F}^c(p_2)\) with the property that \(h^u_{p_1,p_2}(x) \in \mathcal{F}^u(p_1) \cap \mathcal{F}^c(p_2)\).
\end{definition}

\end{figure}
\end{frame}

\begin{frame}
\begin{theorem}
  Let \(f \colon X \to X\) be a diffeomorphism with compact center leaves and such that the quotient \(f_c \colon X/\mathcal{F}^c \to X/\mathcal{F}^c\) is hyperbolic. Let \(\mu\) be an ergodic invariant measure for \(f\), and also suppose \(f\) has global unstable holonomies. Then, \(\lambda^c(\mu) \le 0\) implies that the disintegration \(\set{\mu^c_x}_{x \in M}\) is invariant by unstable holonomies, meaning that there exists a full measure subset \(E \subseteq X\) in which \((h^u_{p_1,p_2})_*\mu_{p_1}^c = \mu^c_{p_2}\) for all \(p_1,p_2 \in E\) that are in the same \(\mathcal{F}^{cu}\) leaf.
\end{theorem}
\end{frame}

\begin{frame}
  \frametitle{A trick in a box}
  Assume we work locally in a leaf of \(\mathcal{F}^{cu}\), parametrized by a box \(B = [0,1]^{u} \times [0,1]^{c}\).
  \begin{figure}
    \centering
    \includegraphics[page=1,scale=0.6]{figures/invp}
  \end{figure}
  Each disintegration \(\set{\mu^u_p}_{p \in B}\) and \(\set{\mu^c_p}_{p \in B}\) is supported in vertical or horizontal sections.

  Fubini implies
  \begin{equation*}
    \set{\mu^c_p}_{p \in B} \text{ invariant} \iff
    \forall p,\;\pi^c_*\mu^c_p = \pi^c_*\mu^{cu} \iff
    \forall p,\;\pi^u_*\mu^u_p = \pi^u_*\mu^{cu}.
  \end{equation*}
\end{frame}

\begin{frame}
  \frametitle{From the box to a leaf}
  \begin{itemize}
  \item Denote by \(\pi \colon X \to X/\mathcal{F}^c\) the quotient map, and let \(\set{\mu^c_p}_{p \in X}\) be the disintegration of \(\mu\) along the (measurable) foliation \(\mathcal{F}^c\).

  \item Since \(f_c \colon X/\mathcal{F}^c \to X/\mathcal{F}^c\) is hyperbolic, the quotient also has stable and unstable foliations, which we denote by \(W^s\) and \(W^u\) respectively. Let \(\nu = \pi_*\mu\) be the pushforward measure on \(X/\mathcal{F}^c\).

  \item We will work with a partition \(\xi_c\) of \(X/\mathcal{F}^c\) whose elements are compact sets and which is finer than the \(W^u\) foliation. We have an associated disintegration \(\set{\nu^u_x}_{x \in X/\mathcal{F}^c}\) of \(\nu\).

  \item Denote \(\xi^{cu} = \pi ^{-1}(\xi_c)\), which by construction is finer than the foliation \(\mathcal{F}^u\). It is also disintegrates \(\mu\) to a family \(\set{\mu^{cu}_p}_{p \in X}\).
  \end{itemize}
  \begin{lemma}\label{lem:proj-implies-invariance}
    For all \(x \in E_3\), if \(\pi_* \mu^u_p = \nu_{\pi(p)}^u\) for \(\mu_x^{cu}\)-almost all \(p\), then there is a full \(\mu_x^{cu}\)-measure set \(E_4(x) \subseteq \xi^{cu}(x)\) where \((h^u_{q_1,q_2})_*\mu^c_{q_1} = \mu^c_{q_2}\) for all \(q_1,q_2 \in E_4(x)\).
  \end{lemma}
\end{frame}

\begin{frame}
  \frametitle{Jensen's inequality}
  \begin{figure}
    \centering
    \includegraphics[page=2,scale=0.6]{../figures/invp}
  \end{figure}
  Fix \(x \in X\), and let \(\mathcal{P}\) be a partition of finite entropy for \(\mu^{cu}_x\). We have 
  \begin{equation*}
    H_{\mu_x^{cu}}(\mathcal{P}) = \sum_{P \in \mathcal{P}} \phi(\mu_x^{cu}(P)),
  \end{equation*}
  \begin{equation*}
    H_{\mu_x^{cu}}(\mathcal{P} \mid \mathcal{F}^u)
    = \int H_{\mu^{u}_y}(\mathcal{P}) \;d\mu_x^{cu}(y)
    = \int \sum_{P \in \mathcal{P}} \phi(\mu_y^{u}(P)) \;d\mu_x^{cu}(y).
  \end{equation*}
\end{frame}

\begin{frame}
  \frametitle{Jensen's inequality}
  \begin{lemma}[Jensen's Inequality]
    Suppose \(\varphi \colon \RR \to \RR\) is a concave function, and let \(\psi \colon \Omega \to \RR\) be a measurable function for a probability space \((\Omega,\mathcal{B},\Prob)\). Then
    \begin{equation*}
      \int \varphi \circ \psi \;d\Prob \le \varphi\rpar[\bigg]{\int \psi \;d\Prob}.
    \end{equation*}
    Moreover, if \(\varphi\) is strictly concave, the equality holds if, and only if, \(\psi\) is constant almost everywhere.
  \end{lemma}
  Applying to \(\Prob = \mu_x^{cu}\), \(\varphi(x) = \phi(x) =  - x\log x\) and \(\psi_P(y) = \mu^u_p(P)\), for each \(P \in \mathcal{P}\)
  \begin{equation*}
    \int \phi(\mu_p^u(P))\;d\mu_x^{cu}(p)\le \phi\rpar[\bigg]{\int \mu_p^u(P)\;d\mu_x^{cu}(p) } = \phi(\mu_x^{cu}(P)),
  \end{equation*}
  so summing over \(P \in \mathcal{P}\),
  \begin{equation*}
    H_{\mu_x^{cu}}(\mathcal{P}\mid \mathcal{F}^u)\le H_{\mu_x^{cu}}(\mathcal{P})
  \end{equation*}
  with equality if, and only if, \(p \mapsto \mu^u_p(P)\) is constant for \(\mu_x^{cu}\)-every \(p\) and all \(P\).
\end{frame}

\begin{frame}
  \frametitle{Jensen's inequality}
  \begin{lemma}\label{lem:almost-there}
  For \(\mu\)-almost every \(x\), if \(\mathcal{P} = \pi ^{-1}(\xi_c)\) is a partition with finite entropy for \(\mu^{cu}_x\) and \(H_{\mu_x^{cu}}(\mathcal{P}\mid \mathcal{F}^u) = H_{\mu_x^{cu}}(\mathcal{P})\), then for \(\mu^{cu}_x\)-almost all \(p\), \(\pi_*\mu^u_p\) and \(\nu^u_p\) coincide in the \(\sigma\)-algebra \(\sigma(\xi_c)\).
\end{lemma}
\begin{lemma}\label{cor:gen-implies-inv}
  Fix \(\xi_c\) and suppose that \(\sigma(\bigcup_n f_c^{-n}(\xi_c))\) is finer than the Borel algebra of \(X/\mathcal{F}^c\). Let \(\mathcal{P}_n = \pi ^{-1}(f_c^{-n}(\xi_c))\). For all \(x \in E_3\), if \(H_{\mu_x^{cu}}(\mathcal{P}_n\mid \mathcal{F}^u) = H_{\mu_x^{cu}}(\mathcal{P}_n)\) for all \(n\), then there exists a full \(\mu_x^{cu}\)-measure set \(E_4(x)\), such that for all \(p_1, p_2 \in E_4(x)\), \((h^u_{p_1,p_2})_*\mu^c_{p_1} = \mu^c_{p_2}\).
\end{lemma}
\begin{lemma}\label{cor:gen-implies-inv-ae}
  Suppose that \(\sigma(\bigcup_n f_c^{-n}(\xi_c))\) is finer than the Borel algebra of \(X/\mathcal{F}^c\), and \(H_{\mu_x^{cu}}(\mathcal{P}_n\mid \mathcal{F}^u) = H_{\mu_x^{cu}}(\mathcal{P}_n)\) for all \(n\) and \(\mu\)-almost every \(x\). Then, there is a full \(\mu\)-measure set \(E_5 \subseteq X\) such that for all \(x,y \in E_5\), if \(y \in \xi^{cu}(x)\), then \((h^u_{x,y})_*\mu^c_x = \mu^c_y\).
\end{lemma}
\end{frame}

\begin{frame}
  \frametitle{Patching the leaves together}
  We have a condition to showing invariance in the same \(\xi^{cu}\)-plaque (element of the partition). How to make that into a condition for the \emph{entire \(\mathcal{F}^{cu}\) leaf}?\\[1em]

  Assume
  \begin{hypothesis}
    For \(\mu\)-almost every \(x\), \(\xi^u(x) = \xi^{cu}(x) \cap \mathcal{F}^u(x)\) is a neighborhood of \(x\) in \(\mathcal{F}^u(x)\).
  \end{hypothesis}
  Then
  \begin{lemma}
    There exists \(\delta > 0\) such that the set
    \begin{equation*}
      S = S_{\delta} = \set{x \in X \where d^u(x, X \setminus \xi^{u}(x)) > \delta}
    \end{equation*}
    has positive measure.
  \end{lemma}
\end{frame}

\begin{frame}
  \begin{lemma}\label{cor:cond-invprin}
  Suppose that \(\sigma(\bigcup_n f_c^{-n}(\xi_c))\) is finer than the Borel algebra of \(X/\mathcal{F}^c\), and \(H_{\mu_x^{cu}}(\mathcal{P}_n\mid \mathcal{F}^u) = H_{\mu_x^{cu}}(\mathcal{P}_n)\) for all \(n\) and \(\mu\)-almost every \(x\). Then, there is a full \(\mu\)-measure set \(E \subseteq X\) such that for all \(x,y \in E\), if \(y \in \mathcal{F}^{u}(x)\), then \((h^u_{x,y})_*\mu^c_x = \mu^c_y\).
\end{lemma}
\begin{proof}
    Choose a full-measure \(E\) such that
  \begin{enumerate}
  \item \(\mu^c_{f^k(x)} = f^k_*\mu^c_x\) for all \(k \in \ZZ\), for all \(x \in E\) (by invariance of \(\mu\));
  \item \(f^{-n}(x) \in S\) for infinitely many \(n > 0\), for all \(x \in E\) (by ergodicity);
  \item \(E \subseteq \bigcap_{k \in \NN} f^{k}(E_5)\), where \(E_5\) is the full \(\mu\)-measure set where invariance holds in the plaques.
  \end{enumerate}
  Let \(x,y \in E\) such that \(y \in \mathcal{F}^u(x)\). We know \(d^u(f^{-n}(x),f^{-n}(y)) \to 0\), so there exists some \(N\) for which \(d^u(f^{-N}(x),f^{-N}(y)) < \delta\) and \(f^{-N}(x) \in S\). By definition of \(S\), this implies \(f^{-N}(y) \in \xi^{cu}(f^{-N}(x))\). Since \(x, y \in f^N(E_5)\),
  \[f^{-N}_*(h^u_{x,y})_*\mu^c_x = (h^u_{f^{-N}(x),f^{-N}(y)})_*\mu^c_{f^{-N}(x)} = \mu^c_{f^{-N}(y)} = f^{-N}_*\mu^c_y.\]
  Now, apply \(f^N_*\) to both sides to get \((h^u_{x,y})_*\mu^c_x = \mu^c_y\).
\end{proof}
\end{frame}

\begin{frame}
  \frametitle{Adapted partitions and unstable entropy}
\begin{definition}[Adapted partition]
  We say a measurable partition \(\xi\) is \(\mu\)-adapted to a foliation \(\mathcal{F}\) if \(\xi \prec f^{-1}(\xi)\) and there exists \(\varepsilon > 0\) such that for \(\mu\)-\almostevery \(x \in M\),
  \begin{enumerate}
  \item\label{item:2} \(\xi(x) \subseteq \mathcal{F}(x)\cap B_{\varepsilon}(x)\) and
  \item\label{item:4} \(\xi(x)\) is a neighborhood of \(x\) inside \(\mathcal{F}(x)\).
  \end{enumerate}
\end{definition}

\end{figure}
\end{frame}

\begin{frame}
  \frametitle{Adapted partitions and unstable entropy}
  \begin{definition}[Unstable entropy]
    Let \(\xi\) be a partition \(\mu\)-adapted to the unstable foliation \(\mathcal{F}^u\). The \emph{unstable entropy} is defined as
    \begin{equation*}
      h_{\mu}^u(f) =
      h_{\mu}(f, \xi) =
      H_{\mu}(f ^{-1}(\xi) \mid \xi) =
      \int H_{\mu_x^u}(f^{-1}(\xi))\;d\mu(x),
    \end{equation*}
    where \(\set{\mu^u_x}_{x \in X}\) is the decomposition of \(\mu\) with respect to \(\xi\).
  \end{definition}
  \begin{lemma}
    The unstable entropy does not depend on the choice of adapted partition \(\xi\).
  \end{lemma}
\end{frame}

\begin{frame}
  \frametitle{Construction of adapted partitions}
  \begin{definition}
    We say a set \(R \subseteq X/\mathcal{F}^c\) is a \emph{proper rectangle} if \(R\) is closed, \(R = \closure{\interior{R}}\) and \([x,y] \in R\) whenever \(x, y \in R\). The bracket \([x,y]\) is defined as the unique point in \(W^s_{\varepsilon}(x)\cap W^u_{\varepsilon}(y)\) for all \(x,y\) such that \(d(x,y) < \delta\) for some \(0 < \delta < \varepsilon\). For that to make sense, we also require that \(\diam R < \delta\).
  \end{definition}

  \begin{definition}
    A \emph{Markov partition} is a finite partition \(\mathcal{R} = \set{R_1,\dots,R_m}\) of \(X/\mathcal{F}^c\) by proper rectangles such that for all \(x \in X\) with \(x, f(x) \notin \bigcup_{i = 1}^m \partial R_i\),
    \begin{enumerate}
    \item\label{item:6} \(W^u(f(x)) \cap \mathcal{R}(f(x)) \subseteq f(W^u(x)\cap \mathcal{R}(x))\) and
    \item \(W^s(f(x)) \cap \mathcal{R}(f(x)) \supseteq f(W^s(x)\cap \mathcal{R}(x))\).
    \end{enumerate}
  \end{definition}
\end{frame}

\begin{frame}
  \frametitle{Construction of adapted partitions}
  Given a Markov partition \(\mathcal{R}\), we define \(\xi_c(x) = W^u(x) \cap \mathcal{R}(x)\). Now, construct \(\xi^{cu} = \pi^{-1}(\xi_c)\) and \(\xi^u = \xi^{cu} \vee \mathcal{F}^u\) as before. We claim \(\xi = \xi^{u}\) is \(\mu\)-adapted to \(\mathcal{F}^u\). Indeed, we have \(\xi^{u} \prec f^{-1}(\xi^{u})\), and outside the zero-measure set \(\pi ^{-1}\rpar*{\bigcup_{i = 1}^m \partial R_i}\), we have \[\xi^{u}(x) = \mathcal{F}^u(x) \cap \pi^{-1}(W^u(\pi(x)) \cap \mathcal{R}(\pi(x))),\]
  so \(\xi^{u}(x)\) is a neighborhood of \(x\) in \(\mathcal{F}^u(x)\) of small diameter because \(\pi(x) \in \interior \mathcal{R}(\pi(x))\).
\end{frame}

\begin{frame}
  \frametitle{Proof of the invariance principle}
  By the last results,
  \begin{align*}
    h_{\mu}^u(f)
    = \int H_{\mu_{x}^u}(f^{-1}(\xi^u))\;d\mu(x)
    &= \int H_{\mu_{x}^u}(f^{-1}(\xi^{cu}))\;d\mu(x)\\
    &= \int H_{\mu_{x}^{cu}}(f^{-1}(\xi^{cu}) \mid \mathcal{F}^u)\;d\mu(x)\\
    &\le \int H_{\mu^{cu}_x}(f^{-1}(\xi^{cu}))\;d\mu(x)\\
    &=  \int H_{\nu^{u}_{\pi(x)}}(f_c^{-1}(\xi_c))\;d\mu(x)\\
    &=  \int H_{\nu^{u}_{y}}(f_c^{-1}(\xi_c))\;d\nu(y)\\
    &= h^u_{\nu}(f_c) = h_{\nu}(f_c)
  \end{align*}
  with equality if, and only if, \(H_{\mu_{x}^{cu}}(f^{-1}(\xi^{cu}) \mid \mathcal{F}^u) = H_{\mu^{cu}_x}(f^{-1}(\xi^{cu}))\) for \(\mu\)-almost every \(x\).
\end{frame}

\begin{frame}
  \frametitle{Proof of the invariance principle}
  The inequality above equally applies to \(f = f^n\), and since the diameter of \(f^{-n}(\xi_c)\) shrinks to zero, \(\sigma\rpar*{\bigcup_nf^{-n}(\xi_c)}\) is finer than the Borel algebra.\\[1em]

  Now, since \((f^n_c, \nu)\) is a factor of \((f^n,\mu)\), we have \(h_{\nu}(f_c^n) \le h_{\mu}(f^n)\). If \(\lambda^c(\mu) \le 0\), then the Ledrappier-Young formula tells us that \(h_{\mu}^u(f^n) = n h_{\mu}^u(f) = n h_{\mu}(f) = h_{\mu}(f^n)\), so we have equality in the equation above.\hfill$\square$
  \begin{theorem}[Ledrappier-Young formula]\label{thm:leddrappier-young}
    Let \(\lambda_1 > \cdots > \lambda_u > 0\) be the distinct positive Lyapunov exponents of \(f\) for \(\mu\), and let \(\delta_i\) be the dimension of \(\mu\) on \(W^i\)-manifolds. Then, for all \(1 \le i \le u\) there exist real numbers \(0 \le \gamma_i \le \dim V^i - \dim V^{i - 1}\) such that
    \begin{equation*}
      h_{\mu}(f) = \sum_{i \le u} \lambda_i\gamma_i.
    \end{equation*}
  \end{theorem}
  
\end{frame}
