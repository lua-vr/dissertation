\begin{frame}
  \frametitle{A nice definition}
  A dynamical system is...\\[2em]
  \[\text{a functor } M \to \mathcal{C}\]\\[2em]
  
  where \(M = \set{\star}\) is a monoid (seen as an one-object category) and \(\mathcal{C}\) is a category with any objects and compositions you like.\\[3em]

  (for more about that, see \Textcite{mossCategorytheoreticProofErgodic2023}).
\end{frame}

\begin{frame}
  \frametitle{Ergodic theory}
  The study of dynamical systems for which...
  \[\mathcal{C} = \text{ category of measure spaces and measure-preserving maps}\]
  \begin{theorem}\label{thm:birkhoff}
    Let  \(\mu\) be an invariant probability measure and \(\phi \colon X \to \mathbb{R}\) an integrable function. Then, for \(\mu\)-almost every \(x \in X\),
    \begin{equation}\label{eq:birkhoff}
      \frac{1}{n}\sum_{i = 0}^{n - 1} \phi (f^i(x)) \to \CE[\mu]{\phi, \mathcal{I}}(x),
    \end{equation}
    where \(\mathcal{I}\) is the sub-\(\sigma\)-algebra of (strictly) invariant sets.
  \end{theorem}
  \begin{definition}
    We say that a measure \(\mu\) is \emph{ergodic} for \(f\) if the algebra \(\mathcal{I}\) is trivial modulo \(\mu\): for every \(A \in \mathcal{I}\), either \(\mu(A) = 0\) or \(\mu(A^{\complement}) = 0\).
  \end{definition}
\end{frame}

\begin{frame}
  \frametitle{Empirical measures}
  Given \(f \colon X \to X\) and a point \(x \in X\), we can define a sequence of \emph{empirical measures}
  \[\Emp[n](x) = \frac{1}{n} \sum_{i = 0}^{n - 1} \delta_{f^i(x)},\]
  or in other words, for all \(\phi \colon X \to \RR\),
  \begin{equation*}
    \int \phi \;d(\Emp[n](x)) = \frac{1}{n}\sum_{i = 0}^{n - 1} \phi(f^i(x)).
  \end{equation*}
  \begin{question}
    Given \(x\), do the \(e_n(x) \weakto \mu\) weakly for some \(\mu\)?
  \end{question}
  \begin{question}
    Given \(\mu\), what are the \(x \in X\) for which \(e_n(x) \weakto \mu\)?
  \end{question}
\end{frame}

\begin{frame}
  \frametitle{Basins}
  The set \(\mathcal{B}(\mu) \subseteq X\) of points for which \(e_n(x) \weakto \mu\) is called the \emph{basin} of \(\mu\).\\[2em]

  What do the basins generally look like?\\[2em]

  Conjecture (Palis): for a \(C^r\)-dense set of diffeomorphisms \(f \colon M \to M\) of a compact manifold \(M\), \(f\) should have a finite number of physical measures (measures with basins of positive volume), such that their basins cover the entire manifold up to a subset of zero volume.\\[1em]

  (See e.g.\ \citetitle{pujalsPeixotosTheoremPaliss2011}.)
\end{frame}

\begin{frame}
  \frametitle{Overview}
  This project reports on interesting phenomena related to empirical measures and their basins.\\[1em]
  \begin{enumerate}
  \item On the first chapter, we analyze a family of skew products by \textcite{bonifantSchwarzianDerivativesCylinder2008} where we see either intermingled basins or a unique asymptotic measure depending on the Schwarzian derivative of the fibers. For that, we also show the existence of a graph that separates the basins.
  \item We then present an important theorem of partially hyperbolic dynamics, called the Invariance Principle. We discuss how this theorem implies some conditions on the Lyapunov exponent of the measure supported on the graph, and on the absolute continuity of the asymptotic measure.
  \item Lastly, we look at an important open question about the empirical measures of diffeomorphisms, namely, the existence of robust historical behavior. We show this occurs non-robustly for skew products when some conditions ensure a kind of convergence to the Brownian motion along the fibers.
  \end{enumerate}
\end{frame}

\section{Schwarzian derivatives and cylinder maps}

\begin{frame}
  \frametitle{Schwarzian derivatives and cylinder maps}
  \textcite{bonifantSchwarzianDerivativesCylinder2008}:
  \begin{gather*}
    F \colon X \times I \to X \times I\\
    F(x,y) = (g(x), f_x(y))
  \end{gather*}
  where the \(f_x \colon I \to I\) are orientation-preserving \(C^3\) diffeomorphisms.\\[1em]

  The behavior of the basins of \(\mu_0 = \mu \otimes \delta_0\) and \(\mu_1 = \mu \otimes \delta_1\) depend on the sign of the \emph{Schwarzian derivative} of the \(f_x\).
\end{frame}

\begin{frame}
  \frametitle{Dividing by zero}
  Consider a field \(\mathbb{K}\). We can add a point \(\infty\) at infinity to get \(P^1(\mathbb{K})\), which has the same operations except that now \(0^{-1} = \infty\), \(\infty ^{-1} = 0\), \(x \cdot \infty = \infty\) if \(x \neq 0\), and \(x + \infty = \infty\) if \(x \neq \infty\).\\[1em]

  \(P^1(\mathbb{K})\) has a special symmetry group given by an action of \(\begin{bsmallmatrix}a&b\\c&d\end{bsmallmatrix} \in \PGL_2(\mathbb{K})\):
  \begin{equation*}
    \gamma_{a,b,c,d}(z) = \frac{az + b}{cz + d}.
  \end{equation*}
  What is the fractional linear \(\rho\) for which \(\rho(x_1) = \infty\), \(\rho(x_2) = 0\), and \(\rho(x_3) = 1\)?
  \begin{definition}[Cross ratio]
    \begin{equation*}
      \rho(x_0,x_1,x_2,x_3) = \frac{(x_2 - x_0)(x_3 - x_1)}{(x_1 - x_0)(x_3 - x_2)}.
    \end{equation*}
  \end{definition}
  \begin{definition}[Poincaré metric]
    The quantity \(d_H(x,y) = \log{\abs{\rho(0,x,y,1)}}\) defines a metric on \((0,1) \subseteq I\).
  \end{definition}
\end{frame}

\begin{frame}{Cross ratio and Schwarzian derivative}
  \begin{definition}
    Let \(\f \colon I \to I'\) be a \(\Cs[d = 3]\) diffeomorphism between two intervals of the real line.
    The \emph{Schwarzian derivative} of \(\f\) is the function
    \begin{equation*}
      \SchD\f = \frac{\f[der = 3]}{\f[der = 1]} - \frac{3}{2}\rpar*{\frac{\f[der = 2]}{\f[der = 1]}}^2.
    \end{equation*}
    We write \(\SchD f \prec 0\) if \(\SchD f < 0\) in an open dense set, and analogously \(\SchD f \succ 0\).
  \end{definition}
  \begin{lemma}\label{lem:sch-cross-increase}
    If \(f \colon I \to I'\) is \(C^3\), then \(\SchD f \prec 0\) if, and only if, \(f\) increases cross ratios, in the sense that \[\rho(f(y_{1}),f(y_2),f(y_3),f(y_4)) > \rho(y_{1},y_2,y_3,y_4)\] for all \(y_1 < y_2 < y_3 < y_{4}\).
  \end{lemma}

\begin{frame}
  \frametitle{Properties of the Schwarzian}
  \begin{proposition}\label{prop:schwarzian}
    The Schwarzian derivative has the following properties:
    \begin{enumerate}
    \item\label{prop:schwarzian:item:1}
      If \(\mathcal{S}f \succ 0\) and \(\mathcal{S}g \succ 0\), then \(\mathcal{S}(f \circ g) \succ 0\), and similarly for \(\prec 0\). Moreover, if \(\mathcal{S}f \equiv 0\), then \(\SchD{f \circ g} = \SchD g \cdot (g')^2\), and if \(\SchD g \equiv 0\), then \(\SchD{f \circ g} = (\SchD f) \circ g\).
    \item If \(\mathcal{S}f \prec 0\), then \(\mathcal{S}f^{-1} \succ 0\). 
    \item\label{prop:schwarzian:item:3} \(\SchD f \prec 0\) if, and only if, \(\varphi(y) = 1/ \sqrt{\lvert f'(y) \rvert}\) is strictly convex.
    \item \(\SchD f \equiv 0\) if, and only if, \(f\) is a fractional linear transformation.
    \end{enumerate}
  \end{proposition}
\end{frame}

\begin{frame}
  \frametitle{Transversal Lyapunov exponents}
  Let \(\mu\) ergodic for \(g \colon X \to X\), and \(\mathcal{A}_i = X \times \set{i}\) for \(i = 0,1\).
  \begin{definition}[Transversal Lyapunov exponent]
    \begin{equation*}
      \Lyap(\mathcal{A}_i) = \int \log f_{x}'(i) \;d \mu(x).
    \end{equation*}
  \end{definition}
  \begin{lemma}\label{lem:lyap-basin-posmeas}
    If the family \(\{ f_x' \}_{x \in X}\) is equicontinuous at \(i \in \{ 0,1 \}\) and \(\Lyap(\mathcal{A}_i) < 0\), then the basin of \(\mu_i\) has positive \(\mu \otimes \Leb\) measure.  Conversely, if \(\Lyap(\mathcal{A}_i) > 0\), then the corresponding basin has null measure.
  \end{lemma}
  \begin{corollary}\label{cor:schwarz-basin-pos-meas}
    If \((x,y) \mapsto \SchD f_x(y)\) is continuous and \(\SchD f_x \prec 0\) for \(\mu\)-\almostevery \(x \in X\), then \(\Lyap(\mathcal{A}_0) + \Lyap(\mathcal{A}_1) < 0\).
    In particular, at least one of the basins has positive measure.
  \end{corollary}
  
\end{frame}

\begin{frame}
  \frametitle{Intermingled basins}
  \begin{definition}
    Let \(M\) be a space with a reference measure \(\mu\). For a set of measures \(\set{\nu_i}_i\) invariant for \(f \colon M \to M\), we say that their basins are intermingled if for each nonempty open \(U \subseteq X\), \(\mu(\Basin{\nu_i} \cap U) > 0\) for all \(i\).
  \end{definition}
\end{frame}
\begin{frame}
  \frametitle{Intermingled basins}
  Remember: \(F(x,y) = (g(x),f_x(y))\).\\[1em]
  Assume:
  \begin{hypothesis}\label{hip:basins-support-nonempty}
    The basins \(\Basin{\mu_{0}}\) and \(\Basin{\mu_{1}}\) have positive volume \(\mu \otimes \Leb\).
  \end{hypothesis}

  \begin{hypothesis}\label{hip:elevators}
    The map \(g\) is topologically exact, in the sense that for each nonempty open set \(U\), there exists \(n\) for which \(g^n(U) = X\). Moreover, there exists two expanding periodic points \(x^+\) and \(x^-\) in \(X\), of periods \(n^+, n^-\), such that for every \(y \in (0,1)\), \(f_{x^+}^{n^+}(y) > y\) and \(f_{x^-}^{n^-}(y) < y\).
  \end{hypothesis}
\end{frame}

\begin{frame}
  \frametitle{Intermingled basins}
  \begin{theorem}
    Under the two hypothesis, the basins of \(\mu_0\) and \(\mu_1\) are intermingled.
  \end{theorem}
  \begin{proof}[Proof sketch]
    If we define \(\beta_i(A) = m(A \cap \Basin{\mu_i})\), it suffices to show \(\supp \beta_i = X \times I\) for \(i = 0,1\).\\[1em]

    Consider \(i = 0\). The first hypothesis ensures there is \((x_0,y_0) \in \supp \beta_0\) with \(y_0 \in (0,1)\). The second ensures there is a backward orbit \(\cdots \mapsto x_{-2} \mapsto x_{-1} \mapsto x_0\) with \(x_{-n}\to \Orbit{x^-}\). By closedness and invariance of \(\supp \beta_0\), \((x^-, 1) \in \supp \beta_0\). The preimages of \((x^-, 1)\) are dense in \(\mathcal{A}_1\), so \(\mathcal{A}_1 \subseteq \supp \beta_0\). But \(\supp \beta_0\) is made of vertical segments starting at \(\mathcal{A}_0\), therefore we must have \(\supp \beta_0 = X \times I\).
  \end{proof}
\end{frame}

\begin{frame}
  \frametitle{Negative Schwarzian}
  Is there something between the two basins?
  \begin{theorem}\label{thm:neg-schwarz-sep-graph}
    If \((x,y) \mapsto \SchD f_x(y)\) is continuous and \(\SchD f_x \prec 0\) almost everywhere in \(x \in X\), then there exists a measurable function \(\sigma \colon X \to I\) defined almost everywhere such that
    \begin{equation*}
      (x,y) \in \Basin{\mu_0} \text{ whenever } y < \sigma(x) \quad \text{and} \quad (x,y) \in \Basin{\mu_1} \text{ whenever }y > \sigma(x).
    \end{equation*}
    Consequently, the union \(\Basin{\mu_0} \cup \Basin{\mu_1}\) has total \(\mu \otimes \Leb\) measure.
  \end{theorem}
  \Textcite[Theorem 3.2]{bonifantSchwarzianDerivativesCylinder2008} also assume both \(\Basin{\mu_0}\), \(\Basin{\mu_1}\) have positive volume, and leave open the question of whether this is necessary in a footnote. We showed this is not necessary.
\end{frame}
\begin{frame}
  \frametitle{Negative Schwarzian}
  \begin{proofs}[Proof sketch]
    Start defining
    \begin{equation*}
      \xi_0(x) = \sup \{ y \in I : (x,y) \in \Basin{\mu_0} \}, \qquad \xi_1(x) = \inf \{ y \in I : (x,y) \in \Basin{\mu_1} \}.
    \end{equation*}
    It suffices to show \(A = \set{ x \in X : \xi_0(x) \neq \xi_1(x) }\) does not have positive measure.\\[1em]

    By a past result, we can assume \(\xi_1(x) < 1\) for \(\mu\)-almost every \(x\). Thus, given \(k\), there exists \(\delta_k < 1\) such that
    \begin{equation}\label{eq:time-small}
      \lim_{n \to \infty} \frac{1}{n} \# \set{ 0 \leq i < n \where \xi_1(g^i(x)) < \delta_k} > 1 - 1/k
    \end{equation}
    for almost every \(x\).\\[1em]

    Suppose there were \(x_0 \in A\) for which the above was valid, and let \(y_0 \in (\xi_0(x_0), \xi_1(x_0))\). By definition, \(e_n(x_0,y_0)\) converges to neither \(\mu_0\) nor \(\mu_1\), so by compactness, the sequence of empirical measures has a limit point \(\nu\) with \(\supp \nu \not\subset X \times \set{0}\).
  \end{proofs}
\end{frame}
\begin{frame}
  \frametitle{Negative Schwarzian}
  \begin{proof}[Proof sketch (cont.)]
    Therefore, there exists \(\varepsilon > 0\) and \(k\) for which \(\nu(X \times \interval[open left]{\varepsilon}{1}) > 1/k\). Since \(\nu\) is a limit point, take a subsequence \(n_j \to \infty\) with
    \begin{equation}\label{eq:time-big}
      \liminf_{j \to \infty} \frac{1}{n_j} \# \set{ 0 \leq i < n_j \where f_{x_0}^i(y_0) > \varepsilon} > 1/k.
    \end{equation}
    In other words, we have just bounded \(f_{x_0}^i(y_0)\) in the interval \((\varepsilon, \delta_k)\) for a positive lower frequency of \(i\).\\[1em]

    The sequence \(r_n = \rho(0, f_{x_0}^n(y_0), f_{x_0}^n(\xi_1(x_0)), 1)\) is well defined and increasing. Thus either \(r_n \to \infty\) or \(r_n \to r\).\\[1em]

    But it's impossible for \(r_n \to \infty\) because we have just shown
    \[0 < \varepsilon < f^n_{x_0}(y_0) < f^n_{x_0}(\xi_1(x_0)) < \delta_k < 1\]
    for arbitrarily large \(n\). And it's impossible for \(r_n \to r\), as it would imply in the existence of an invariant graph strictly between \(\xi_0\) and \(\xi_1\). Thus, \(x_0\) does not exist in the first place.
  \end{proof}
\end{frame}

\begin{frame}
  \frametitle{Positive Schwarzian}
  \begin{theorem}\label{thm:pos-schwarz-asymptotic}
    If \(\SchD f_x \succ 0\) almost everywhere, then \(F\) has a unique asymptotic measure \(\nu\) for \(\mu \otimes \Leb\)-almost every point. Furthermore, for each \(i \in \{ 0,1 \}\), \(\Lyap(\mathcal{A}_i) > 0\) implies \(\nu(\mathcal{A}_i) = 0\).
  \end{theorem}
  Generalizing \textcite[Lemma 5.4]{bonifantSchwarzianDerivativesCylinder2008}, we do not assume both \(\Lyap(\mathcal{A}_i)\) are strictly positive.
  \begin{proofs}[Proof sketch]
    Let \(\pi \colon (\eX, \eg, \emu) \to (X, g, \mu)\) be the invertible extension of \(g\), and 
    \begin{equation*}
      \eF(\ex, y) = (\eg(\ex), f_{\pi(\ex)}(y)).
    \end{equation*}
    
    The inverse \(\tilde{F}^{-1}\) satisfies \(\SchD f_{\pi(\ex)}^{-1} \prec 0\) \(\tilde \mu\)-almost everywhere, so it has a separating graph \(\sigma\) by the last theorem.
  \end{proofs}

\end{frame}

\begin{frame}
  \begin{proof}[Proof sketch (cont.)]
    For \(\ex\) in the basin of \(\emu\) and \(y \in (0,1)\), consider  \(r_n = d_H(f_{\pi(\ex)}^n(y), \sigma(\eg^n(\ex)))\). Since \(\SchD f_x \succ 0\), \(r_n > 0\) is decreasing, and therefore converges to \(r_n \to r \ge 0\).\\[1em]

    If \(r > 0\), there would exist another invariant graph with finite distance from \(\sigma\), which is not possible. So \(r = 0\), and \(\tilde{F}^n(\ex,y)\) converges in distance to \(\graph \sigma\). Therefore \[e_n(\ex, y) \to \enu = (\graph \sigma)_*\tilde \mu.\]
    Pushing it to the base via \(\pi\), we have the asymptotic measure \(\nu = \pi_*\tilde \nu\).
  \end{proof}
\end{frame}

\section{Partially hyperbolic dynamics}

\begin{frame}
  \frametitle{Dominated splittings}
  \begin{definition}
    We say \(TM = E_1\oplus \cdots \oplus E_n\) is a \emph{splitting} of \(TM\) if at each \(x \in M\), \(T_xM = E_1(x)\oplus \cdots \oplus E_n(x)\) and \(\dim E_i(x)\) does not depend on \(x\).
  \end{definition}
  \begin{definition}
    We say a splitting \(TM = E_1\oplus \cdots \oplus E_n\) is \emph{dominated} for a diffeomorphism \(f \colon M \to M\) if:
    \begin{enumerate}
    \item the splitting is \(df\)-\emph{invariant}, that is, \(df_x(E_i(x)) = E_i(f(x))\) for all \(i,x\);
    \item there exists constants \(c > 0\) and \(\lambda \in (0,1)\) such that for all \(1 \le i \le n - 1\), \(x \in M\) \(u \in E_i(x) \setminus \set{0}\), \(v \in E_{i + 1}(x) \setminus \set{0}\) and \(k \ge 0\),
      \begin{equation*}
        \frac{\norm{df^k_x(u)}}{\norm{u}} \le c \lambda^k \frac{\norm{df^k_x(v)}}{\norm{v}}.
      \end{equation*}
    \end{enumerate}
  \end{definition}
\end{frame}

\begin{frame}
  \frametitle{Partial hyperbolicity}
  \begin{definition}
    A subbundle \(E \subseteq TM\) is \emph{uniformly contracted} by \(df\) if there exists \(c > 0\) and \(\lambda \in (0,1)\) such that for all \(x \in M\), \(u \in E(x)\) and \(k \ge 0\),
    \begin{equation*}
      \norm{df^k_x(u)} \le c \lambda^k \norm{u}.
    \end{equation*}
  \end{definition}

  \begin{definition}
    Let \(f \colon M \to M\) be a diffeomorphism. We say \(f\) is \emph{partially hyperbolic} if there exists a dominated splitting \(TM = E^s \oplus E^c \oplus E^u\) such that \(E^s\) is uniformly contracted by \(df\) and \(E^u\) is uniformly contracted by \(df ^{-1}\).
  \end{definition}
  \begin{example}
    If \(g \colon M \to M\) is an Anosov diffeomorphism, any skew product diffeomorphism \(F \colon M \times N \to M \times N\) over \(g\) is partially hyperbolic.
  \end{example}
\end{frame}

\begin{frame}
  \frametitle{Dynamical coherence}
  \begin{definition}
    A partially hyperbolic diffeomorphism is \textit{dynamically coherent} if there exists invariant foliations \(\fF^{cs}\) and \(\fF^{cu}\) tangent to \(E^c \oplus E^s\) and \(E^c \oplus E^u\), respectively. In this case, we can also define \(\fF^c = \fF^{cs}\vee \fF^{cu}\).
  \end{definition}
  \begin{example}
    For a skew product diffeomorphism \(F \colon M \times N \to M \times N\) over an Anosov \(g \colon M \to M\), \(g\) has stable and unstable foliations \(W^s\) and \(W^u\), and we have \(\fF^{cs} = \pi_M^{-1}(W^s)\) and \(\fF^{cu} = \pi_M^{-1}(W^u)\). Therefore, the center foliation is composed of sections \(\mathcal{F}^c(x) = \set{x} \times N\).
  \end{example}
\end{frame}

\begin{frame}{Global unstable holonomies}
\begin{definition}
  Consider the quotient map \(\pi \colon X \to X/\mathcal{F}^c\). We say that a dynamically coherent partially hyperbolic diffeomorphism \(f \colon X \to X\) has \emph{global unstable holonomies} if for every \(p_1,p_2 \in X/\mathcal{F}^c\) that are in the same leaf of \(\pi(\mathcal{F}^{u})\), there exists an homeomorphism \(h^u_{p_1,p_2}\colon \mathcal{F}^c(p_1) \to \mathcal{F}^c(p_2)\) with the property that \(h^u_{p_1,p_2}(x) \in \mathcal{F}^u(p_1) \cap \mathcal{F}^c(p_2)\).
\end{definition}

\end{figure}
\end{frame}

\begin{frame}
  \frametitle{Invariance principle}
  \begin{definition}
    We say that the dynamically coherent partially hyperbolic diffeomorphism \(f\) has \emph{compact center leaves} if all the leaves of \(\mathcal{F}^c\) are compact sets.
\end{definition}

\begin{theorem}
  Let \(f \colon X \to X\) be a PH diffeomorphism with compact center leaves and such that the quotient \(f_c \colon X/\mathcal{F}^c \to X/\mathcal{F}^c\) is hyperbolic (Anosov), and suppose \(f\) has global unstable holonomies. Let \(\mu\) be an ergodic invariant measure for \(f\). Then, \(\lambda^c(\mu) \le 0\) implies that the disintegration \(\set{\mu^c_x}_{x \in M}\) is invariant by unstable holonomies, meaning that there exists a full measure subset \(E \subseteq X\) in which \((h^u_{p_1,p_2})_*\mu_{p_1}^c = \mu^c_{p_2}\) for all \(p_1,p_2 \in E\) that are in the same \(\mathcal{F}^{cu}\) leaf.
\end{theorem}
\end{frame}

\begin{frame}
  \frametitle{A trick in a box}
  Assume we work locally in a leaf of \(\mathcal{F}^{cu}\), parametrized by a box \(B = [0,1]^{u} \times [0,1]^{c}\).
  \begin{figure}
    \centering
    \includegraphics[page=1,scale=0.6]{figures/invp}
  \end{figure}
  Each disintegration \(\set{\mu^u_p}_{p \in B}\) and \(\set{\mu^c_p}_{p \in B}\) is supported in vertical or horizontal sections.

  Fubini implies
  \begin{equation*}
    \set{\mu^c_p}_{p \in B} \text{ invariant} \iff
    \forall p,\;\pi^c_*\mu^c_p = \pi^c_*\mu^{cu} \iff
    \forall p,\;\pi^u_*\mu^u_p = \pi^u_*\mu^{cu}.
  \end{equation*}
\end{frame}

\begin{frame}
  \frametitle{Jensen's inequality}
  \begin{figure}
    \centering
    \includegraphics[page=2,scale=0.6]{../figures/invp}
  \end{figure}
  Fix \(x \in X\), and let \(\mathcal{P}\) be a partition of finite entropy for \(\mu^{cu}_x\). We have 
  \begin{equation*}
    H_{\mu_x^{cu}}(\mathcal{P}) = \sum_{P \in \mathcal{P}} \phi(\mu_x^{cu}(P)),
  \end{equation*}
  \begin{equation*}
    H_{\mu_x^{cu}}(\mathcal{P} \mid \mathcal{F}^u)
    = \int H_{\mu^{u}_y}(\mathcal{P}) \;d\mu_x^{cu}(y)
    = \int \sum_{P \in \mathcal{P}} \phi(\mu_y^{u}(P)) \;d\mu_x^{cu}(y).
  \end{equation*}
\end{frame}

\begin{frame}
  \frametitle{Jensen's inequality}
  \begin{lemma}[Jensen's Inequality]
    Suppose \(\varphi \colon \RR \to \RR\) is a strictly concave function, and \(\psi \colon \Omega \to \RR\) is an integrable function for a probability space \((\Omega,\Prob)\). Then
    \begin{equation*}
      \int \varphi \circ \psi \;d\Prob \le \varphi\rpar[\bigg]{\int \psi \;d\Prob}
    \end{equation*}
    and the equality holds if, and only if, \(\psi\) is constant almost everywhere.
  \end{lemma}
  Applying to \(\Prob = \mu_x^{cu}\), \(\varphi(x) = - x\log x\) and \(\psi_P(y) = \mu^u_p(P)\), and summing over \(P \in \mathcal{P}\),
  \begin{equation*}
    H_{\mu_x^{cu}}(\mathcal{P}\mid \mathcal{F}^u)\le H_{\mu_x^{cu}}(\mathcal{P})
  \end{equation*}
  with equality if, and only if, \(p \mapsto \mu^u_p(P)\) is constant for \(\mu_x^{cu}\)-every \(p\) and all \(P\).
  \begin{lemma}\label{lem:almost-there}
    For \(\mu\)-almost every \(x\), if \(\mathcal{P} = \pi ^{-1}(\xi_c)\) is a partition with finite entropy for \(\mu^{cu}_x\) and \(H_{\mu_x^{cu}}(\mathcal{P}\mid \mathcal{F}^u) = H_{\mu_x^{cu}}(\mathcal{P})\), then for \(\mu^{cu}_x\)-almost all \(p\), \(\pi_*\mu^u_p\) and \(\nu^u_p\) coincide in the \(\sigma\)-algebra \(\sigma(\xi_c)\).
  \end{lemma}
\end{frame}

\begin{frame}
  \frametitle{From the box to a leaf}
  \begin{itemize}
  \item Denote by \(\pi \colon X \to X/\mathcal{F}^c\) the quotient map, and let \(\set{\mu^c_p}_{p \in X}\) be the disintegration of \(\mu\) along the (measurable) foliation \(\mathcal{F}^c\).

  \item Since \(f_c \colon X/\mathcal{F}^c \to X/\mathcal{F}^c\) is hyperbolic, the quotient also has stable and unstable foliations, which we denote by \(W^s\) and \(W^u\) respectively. Let \(\nu = \pi_*\mu\) be the pushforward measure on \(X/\mathcal{F}^c\).

  \item We will work with a partition \(\xi_c\) of \(X/\mathcal{F}^c\) whose elements are compact sets and which is finer than the \(W^u\) foliation. We have an associated disintegration \(\set{\nu^u_x}_{x \in X/\mathcal{F}^c}\) of \(\nu\).

  \item Denote \(\xi^{cu} = \pi ^{-1}(\xi_c)\), which by construction is finer than the foliation \(\mathcal{F}^u\). It is also disintegrates \(\mu\) to a family \(\set{\mu^{cu}_p}_{p \in X}\).
  \end{itemize}
  \begin{lemma}
  Suppose that \(\sigma(\bigcup_n f_c^{-n}(\xi_c))\) is finer than the Borel algebra of \(X/\mathcal{F}^c\), and \(H_{\mu_x^{cu}}(\mathcal{P}_n\mid \mathcal{F}^u) = H_{\mu_x^{cu}}(\mathcal{P}_n)\) for all \(n\) and \(\mu\)-almost every \(x\). Then, the disintegration \(\set{\mu^c_x}_{x \in M}\) is invariant by unstable holonomies.
\end{lemma}
  
\end{frame}

\begin{frame}
  \frametitle{Adapted partitions and unstable entropy}
\begin{definition}[Adapted partition]
  We say a measurable partition \(\xi\) is \(\mu\)-adapted to a foliation \(\mathcal{F}\) if \(\xi \prec f^{-1}(\xi)\) and there exists \(\varepsilon > 0\) such that for \(\mu\)-\almostevery \(x \in M\),
  \begin{enumerate}
  \item\label{item:2} \(\xi(x) \subseteq \mathcal{F}(x)\cap B_{\varepsilon}(x)\) and
  \item\label{item:4} \(\xi(x)\) is a neighborhood of \(x\) inside \(\mathcal{F}(x)\).
  \end{enumerate}
\end{definition}

\begin{frame}
  \frametitle{Adapted partitions and unstable entropy}
  \begin{definition}[Unstable entropy]
    Let \(\xi\) be a partition \(\mu\)-adapted to the unstable foliation \(\mathcal{F}^u\). The \emph{unstable entropy} is defined as
    \begin{equation*}
      h_{\mu}^u(f) =
      h_{\mu}(f, \xi) =
      H_{\mu}(f ^{-1}(\xi) \mid \xi) =
      \int H_{\mu_x^u}(f^{-1}(\xi))\;d\mu(x),
    \end{equation*}
    where \(\set{\mu^u_x}_{x \in X}\) is the decomposition of \(\mu\) with respect to \(\xi\).
  \end{definition}
  \begin{lemma}
    The unstable entropy does not depend on the choice of adapted partition \(\xi\).
  \end{lemma}
\end{frame}

\begin{frame}
  \frametitle{Construction of adapted partitions}
  \begin{definition}
    We say a set \(R \subseteq X/\mathcal{F}^c\) is a \emph{proper rectangle} if \(R\) is closed, \(R = \closure{\interior{R}}\) and \([x,y] \in R\) whenever \(x, y \in R\). The bracket \([x,y]\) is defined as the unique point in \(W^s_{\varepsilon}(x)\cap W^u_{\varepsilon}(y)\) for all \(x,y\) such that \(d(x,y) < \delta\) for some \(0 < \delta < \varepsilon\).
  \end{definition}

  \begin{definition}
    A \emph{Markov partition} is a finite partition \(\mathcal{R} = \set{R_1,\dots,R_m}\) of \(X/\mathcal{F}^c\) by proper rectangles such that for all \(x \in X\) with \(x, f(x) \notin \bigcup_{i = 1}^m \partial R_i\),
    \begin{enumerate}
    \item \(W^s(f(x)) \cap \mathcal{R}(f(x)) \supseteq f(W^s(x)\cap \mathcal{R}(x))\) and
    \item \(W^u(f(x)) \cap \mathcal{R}(f(x)) \subseteq f(W^u(x)\cap \mathcal{R}(x))\).
    \end{enumerate}
  \end{definition}
  \begin{lemma}
    Given a Markov partition \(\mathcal{R}\) for \(f_c\), define \(\xi_c(x) = W^u(x) \cap \mathcal{R}(x)\). Let \(\xi^{cu} = \pi^{-1}(\xi_c)\) and \(\xi^u = \xi^{cu} \vee \mathcal{F}^u\). Then, \(\xi^u\) is \(\mu\)-adapted to \(\mathcal{F}^u\).
  \end{lemma}
\end{frame}

\begin{frame}
  \frametitle{Proof of the invariance principle}
  \begin{proofs}
    By the last results,
    \begin{align*}
      h_{\mu}^u(f)
      = \int H_{\mu_{x}^u}(f^{-1}(\xi^u))\;d\mu(x)
      &= \int H_{\mu_{x}^u}(f^{-1}(\xi^{cu}))\;d\mu(x)\\
      &= \int H_{\mu_{x}^{cu}}(f^{-1}(\xi^{cu}) \mid \mathcal{F}^u)\;d\mu(x)\\
      &\le \int H_{\mu^{cu}_x}(f^{-1}(\xi^{cu}))\;d\mu(x)\\
      &=  \int H_{\nu^{u}_{\pi(x)}}(f_c^{-1}(\xi_c))\;d\mu(x)\\
      &=  \int H_{\nu^{u}_{y}}(f_c^{-1}(\xi_c))\;d\nu(y)\\
      &= h^u_{\nu}(f_c) = h_{\nu}(f_c)
    \end{align*}
    with equality if, and only if, \(H_{\mu_{x}^{cu}}(f^{-1}(\xi^{cu}) \mid \mathcal{F}^u) = H_{\mu^{cu}_x}(f^{-1}(\xi^{cu}))\) for \(\mu\)-almost every \(x\).
  \end{proofs}
\end{frame}

\begin{frame}
  \frametitle{Proof of the invariance principle}
  \begin{proofs}[Proof (cont.)]
    The inequality above equally applies to \(f = f^n\), and since the diameter of \(f_c^{-n}(\xi_c)\) shrinks to zero, \(\sigma\rpar*{\bigcup_nf_c^{-n}(\xi_c)}\) is finer than the Borel algebra.\\[1em]

    Now, since \((f^n_c, \nu)\) is a factor of \((f^n,\mu)\), we have \(h_{\nu}(f_c^n) \le h_{\mu}(f^n)\). If \(\lambda^c(\mu) \le 0\), then the Ledrappier-Young formula tells us that \(h_{\mu}^u(f^n) = n h_{\mu}^u(f) = n h_{\mu}(f) = h_{\mu}(f^n)\), so we have equality in the equation above.\hfill$\square$
  \end{proofs}
  \begin{theorem}[Ledrappier-Young formula]\label{thm:leddrappier-young}
    Let \(\lambda_1 > \cdots > \lambda_u > 0\) be the distinct positive Lyapunov exponents of \(f\) for \(\mu\). Then, for all \(1 \le i \le u\) there exist real numbers \(0 \le \gamma_i \le \dim V^i - \dim V^{i - 1}\) such that
    \begin{equation*}
      h_{\mu}(f) = \sum_{i \le u} \lambda_i\gamma_i.
    \end{equation*}
  \end{theorem}
  
\end{frame}

\begin{frame}
  \frametitle{Applications to Bonifant-Milnor}
  \begin{corollary}
    If \(F(x,y) = (g(x),f_x(y))\) displays intermingled basins and has a separating graph \(\sigma \colon X \to I\), then \(\sigma\) is not continuous, while the unstable holonomies are. This means we cannot have \(\lambda^c(\nu)\le 0\), where \(\nu = (\graph \sigma)_*\mu\).
  \end{corollary}
  \begin{question}
    For the positive Schwarzian case, when there exists an asymptotic measure \(\nu\), when are the \(\set{\nu_x^c}_x\) equivalent to the Lebesgue measure on the fibers \(\set{x} \times I\)?
  \end{question}
  \begin{proposition}[{\textcite[Proposition 6.7]{crovisierInvariancePrincipleNoncompact2023}}]
    Let \(f \colon M \to M\) be a partially hyperbolic, quasi-isometric in the center, accessible, center bunched, \(C^2\) diffeomorphism. For every ergodic measure \(\mu\) with local \(cs \times u\)-product structure, full support and satisfying \(h(f,\mu) = h^u(f,\mu) = h^s(f,\mu)\), the center disintegrations \(\set{\mu^c_x}\) are equivalent to the Lebesgue measures along center leaves.
  \end{proposition}
\end{frame}

\section{Historical behavior}

\begin{frame}
  \frametitle{History}
  \begin{definition}
    We say \(f \colon X \to X\) has \emph{historical behavior} for a reference volume measure if for \(x\) in a set of positive volume, the sequence \((e_n(x))_n\) does not converge.
  \end{definition}
  \begin{question}[open]
    Is there a compact manifold \(M\) and a \(C^r\)-open set of diffeomorphisms \(f \colon M \to M\) with historical behavior?
  \end{question}
  This has been asked by \textcite{ruelleHistoricalBehaviourSmooth2001}, and later Floris Takens exposed the problem in~\textcite{takensOrbitsHistoricBehaviour2008}. For this reason and his work in related matters, the open question also became known as \textit{Taken's Last Problem}.
\end{frame}

\begin{frame}
  \frametitle{Brownian motion}
  \begin{definition}
    Let \(\mathcal{C}\) be the product \(\sigma\)-algebra on \(\CWPS\). The Wiener measure \(\WM\) is the unique measure on \((\CWPS, \mathcal{C})\) defined by the following properties: if we denote by \(\WM[t] \colon \CWPS \to \RR\) the evaluation at time \(t\), then
    \begin{enumerate}[(i)]
    \item \(\WM[0] = 0\) almost surely;
    \item for \(t > 0\), \(\WM[t] \sim \Normal[0][t]\);
    \item if \(0 \leq t_1 < t_2 < \dots < t_n\), then the increments \(\WM[t_1], \WM[t_2] - \WM[t_1], \dots, \WM[t_n] - \WM[t_{n - 1}]\) are independent random variables.
    \end{enumerate}
  \end{definition}
\end{frame}

\begin{frame}
  \frametitle{Skorohod space}
  \begin{definition}[Skorohod space]
    The space of functions \([0,1] \to \RR\) whose discontinuities are of the \textit{first kind}, that is, both lateral limits exist and values at a point equal the right limit:
    \begin{equation*}
      \Sko{[0,1]} = \set[par=\Big]{\, f \colon [0,1] \to \RR \where \forall t, \lim_{s \to t^-}x(s) \text{ exists and} \lim_{s \to t^+}x(s) = x(t) \,}.
    \end{equation*}
    
  \end{definition}
  \begin{definition}[Skorohod topology]
    Let \(\Lambda\) be the set of strictly increasing continuous functions from \([0,1]\) to itself. For \(f,g \in \WPS\), we define
    \begin{equation}\label{eq:sko-metric}
      d(f,g) = \inf_{\lambda \in \Lambda} \max\set[par = \big]{ \lVert \lambda - \Id \rVert_{\infty}, \lVert f - g \circ \lambda \rVert_{\infty}}.
    \end{equation}
  \end{definition}
  Since \(d(f,g) \leq \lVert f - g \rVert_{\infty}\), the Skorohod topology is coarser than the uniform topology.
\end{frame}

\begin{frame}
  \frametitle{Weak invariance principle}
  Fix a probability space \((\Omega, \mathcal{F}, \Prob)\) and a sequence \(\xi = (\xi_0,\xi_1,\dots)\) of random variables.\\[1em]

  Define \(\PSum[n,rv=\xi] \colon \Omega \to \RR\), \(\DProc[n,rv=\xi] \colon \Omega \to \WPS\) by
  \[\PSum[n,rv=\xi] = \sum_{i = 0}^{n - 1}\xi_i \qquad \text{and} \qquad\DProc[n,rv=\xi]{\omega}{t} = \frac{\PSum[\floor{nt},rv=\xi]{\omega}}{\sqrt{n}}.\]
  \begin{definition}
    We say \(\xi\) satisfies an \emph{weak invariance principle} if \(\DProc[n,spar,pf]\Prob \weakto \sigma W\) for some \(\sigma > 0\). In other words, for every continuous function \(\varphi \colon \WPS \to \mathbb{R}\),
    \[\int \varphi \circ \DProc[n] \;d\Prob \to \int \varphi \;dW.\]
  \end{definition}
\end{frame}

\begin{frame}
  \frametitle{Weak invariance principle}
  Classic:
  \begin{theorem}[Donsker]
    If \(\xi \subseteq L^2(\Prob)\) is a sequence of \iid variables with mean zero and variance \(1\), then the measures \(\DProc[n,spar,pf] \Prob\) converge weakly to the Wiener measure.
  \end{theorem}
  But there are results more general than that!
\end{frame}

\begin{frame}
  \frametitle{Weak invariance principle}
  \begin{definition}
    A sequence \(\xi = (\xi_1,\xi_2\dots)\) of random variables taking values in \(Y\) is said to be \emph{stationary} if \(\xi_*\Prob\) is an invariant measure for the left shift \(\sigma \colon Y^{\NN} \to Y^{\NN}\).
  \end{definition}
  \begin{definition}
    A sequence \(\xi = (\xi_1,\xi_2\dots)\) of random variables taking values in \(Y\) is said to be \emph{ergodic} if \(\xi_*\Prob\) is an ergodic measure for the left shift \(\sigma \colon Y^{\NN} \to Y^{\NN}\). 
  \end{definition}
  \begin{definition}
    A sequence \(\xi = (\xi_1,\xi_2\dots) \in L^1(\Prob)^{\NN}\) of integrable random variables taking values in \(\RR\)  is said to be a \emph{martingale difference sequence} if for all \(n\),
    \begin{equation*}
      \CE[\Prob]{\xi_{n}, {\sigma(\set{\xi_i \where i < n})}} = 0.
    \end{equation*}
  \end{definition}
   Similarly, we have definitions for two-sided sequences by replacing \(\NN\) by \(\ZZ\).
\end{frame}

\begin{frame}
  \frametitle{Weak invariance principle}
  \begin{theorem}[{\textcite[Theorem 18.3]{billingsleyConvergenceProbabilityMeasures1999}}]
    If \(\xi \subseteq L^2(\Prob)\) is a stationary ergodic martingale difference sequence for which \(\sigma = \norm{\xi_1}_2\) satisfies \(0 < \sigma < \infty\), then \(\DProc[n, spar, pf] \Prob \weakto \sigma W\).
  \end{theorem}
  \begin{definition}
    A sequence \(\xi = (\xi_1,\xi_2\dots)\) of random variables is said to be a \textit{reverse} martingale difference sequence if for all \(n\),
    \begin{equation*}
      \CE[\Prob]{\xi_{n}, {\sigma(\set{\xi_i \where i > n})}} = 0.
    \end{equation*}
  \end{definition}
  \begin{corollary}\label{cor:donsker-martingale}
    If \(\xi \subseteq  L^2(\Prob)\) is a stationary ergodic reverse martingale difference sequence for which \(\sigma = \norm{\xi_1}_2 > 0\), then the measures \({(D^{\xi}_n)}_* \Prob \weakto \sigma W\).
  \end{corollary}

\end{frame}

\begin{frame}[fragile]
  \frametitle{Skew products}
  Return to the dynamical setting where \(F \colon X \times I \to X \times I\) is given by \(F(x,y) = (g(x),f_x(y))\), and \(\mu\) is an ergodic probability for \(g\). Suppose there is a homeomorphism \(\gamma \colon I \to \RR[closure]\) and \(t_x(y) = y + p(x)\) for which
  \[
    \begin{tikzcd}
      I \arrow[d, "\gamma"]\arrow[r, "f_{x}"] & I \arrow[d, "\gamma"]\\
      \RR[closure] \arrow[r,"t_{x}"] & \RR[closure]
    \end{tikzcd}
  \]
  commutes. The dynamic on the fiber is conjugated to a ``random'' walk by steps \(p(g^n(x))\).

  \begin{example}
    If all the fiber maps \(f_x\) satisfy \(\SchD f_x \equiv 0\), then \(f_x(y) = \frac{a_x y}{1 + (a_x - 1)y}\).\\[1em]

    In this case, we can define \(p(x) = \log a_x\) and \(\gamma(y) = \log y - \log (1 - y)\) and we will have \(\gamma \circ f_x = t_x \circ \gamma\).
  \end{example}
\end{frame}

\begin{frame}
  \frametitle{Historical behavior}
  Define \(\xi = (p \circ g^{i})_{i \geq 0}\).
  \begin{theorem}\label{thm:wip-historical}
    If \(\DProc[n,rv = \xi,spar,pf]\mu \weakto \sigma W\) for \(\sigma > 0\), then \(F\) has historical behavior. More specifically, for \(\mu\)-almost every \(x\) and every \(y \in (0,1)\),
    \begin{equation*}\label{eq:emp-closure}
      \set{\mu_0, \mu_1} \subseteq \set{e_n(x,y)}[n \in \NN, ']
    \end{equation*}
    where \(\mu_0 = \mu \otimes \delta_{0}\) and \(\mu_1 = \mu \otimes \delta_{1}\) and the limit points are taken in the weak-\(\ast\) topology.
  \end{theorem}
  We use:
  \begin{lemma}[Forgery Theorem, or Support Theorem]\label{thm:forgery}
    If \(f \in \CWPS\) and \(f(0) = 0\), then every neighborhood of \(f\) has positive Wiener measure.
  \end{lemma}
  (There exists a proof using the arc-sine law instead.)
\end{frame}

\begin{frame}[fragile]
  \frametitle{Historical behavior}
  \begin{proofs}[Proof sketch]
    Let \(\delta > 0\) and \(s_{\delta} \colon [0,1] \to \RR\) be any continuous function such that \(s_{\delta}(0) = 0\) and \(s_{\delta}(t) > 1 + \delta\) for \(t > \delta\).
    \[
      \begin{tikzpicture}[scale=2]
        
        % Parameters for the plot
        \def\del{0.2}

    Let \(U_{\delta} = B_{\delta}(s_{\delta})\) be an open ball of radius \(\delta\) around \(s_{\delta}\) in the uniform topology. We know \(W(U_{\delta}) > 0\), so
    \begin{equation*}
      \liminf_{n \to \infty}(\DProc[n,spar,pf]\mu)(U_{\delta}) \geq \sigma W(U_{\delta}) > 0.
    \end{equation*}
  \end{proofs}
\end{frame}

\begin{frame}
  \begin{proof}[Proof sketch (cont.)]
    Therefore,
    \begin{equation*}
      S_{\delta} = \limsup_{n \to \infty}\, \DProc[n,spar,inv]{U_{\delta}} \subseteq X
    \end{equation*}
    has measure \(\mu(S_{\delta}) > 0\). Since it is invariant, \(\mu(S_{\delta}) = 1\). If \(x \in S_{\delta}\), then there is \(n_k^+ \to \infty\) for which \(D_{n_k}^{\xi}(x) \in U_{\delta}\) for all \(k\). This in turn implies that
    \begin{align*}
      \#\set[par = auto]{ 0 \leq i < n_k^+ \where S_{i}^{\xi}(x) > \sqrt{\textstyle n^+_k} } \geq (1 - \delta)n_k^{+}.
    \end{align*}
    By diagonalization for \(\delta \to 0\), we find \(n^+_k \to \infty\) for which for \(\mu\)-\almostevery \(x\),
    \begin{equation*}
      \frac{1}{n_k^+}\#\set[par = auto]{ 0 \leq i < n_k^+ \where S_{i}^{\xi}(x) > \sqrt{\textstyle n^+_k} } \to 1.
    \end{equation*}
    This will imply
    \[e_{n^+_k}(x,y) \weakto \mu_1\]
    for all \(y \in (0,1)\), and similarly we find \(n^-_k \to \infty\) for which \(e_{n^-_k}(x,y) \weakto \mu_0\).
  \end{proof}

\end{frame}

\begin{frame}
  \frametitle{A decomposition theorem}
  \begin{definition}
    We say \(p \in L^2(\mu)\) has a reverse martingale-coboundary decomposition for \(g \colon X \to X\) if \(p\) can be written as
    \begin{equation*}
      p = q + s \circ g - s,
    \end{equation*}
    where \(q,s \in L^2(\mu)\) and \(\zeta = (q \circ g^i)_{i \geq 0}\) is a reverse martingale difference sequence.
  \end{definition}
  \begin{lemma}\label{thm:martingale-coboundary}
    Suppose \(p \in L^2(\mu)\) has a reverse martingale-coboundary decomposition for \(g\). Let \(\sigma = \norm{q}_2 \). If \(\sigma > 0\), then \(\xi = (p \circ g^i)_{i \ge 0}\) is such that \(\DProc[n,spar,pf]\mu \weakto \sigma W\).
  \end{lemma}
\end{frame}

\begin{frame}
  \frametitle{A decomposition theorem}
  For non-invertible \(g\):
  \begin{theorem}[{\textcite[Theorem 2.11]{vianaStochasticDynamicsDeterministic1997}}]\label{thm:decomposition-existence}
    Suppose \(p \in L^2(\mu)\) is \(\mathcal{F}\)-measurable and \(\mathcal{F}_n = g^{-n}(\mathcal{F})\) are such that
    \begin{equation*}
      \sum_{n = 0}^{\infty} \lVert \CE[\mu]{p, \mathcal{F}_n} \rVert_2 < \infty.
    \end{equation*}
    Then a reverse martingale-coboundary decomposition \(p = q + s \circ g - s\) exists. Moreover, this implies that
    \begin{equation*}
      \int p \;d\mu = \int q \;d\mu = 0
    \end{equation*}
    and we have the equality
    \begin{equation*}
      \sigma ^2 = \lVert q \rVert^2_2 = \lVert p \rVert_2^2 + 2\sum_{n = 1}^{\infty} \int p \cdot p \circ g^n \;d\mu.
    \end{equation*}
  \end{theorem}
  This gives a way to calculate \(\sigma\) in terms of \(p\).
\end{frame}

\begin{frame}[fragile]
  \frametitle{A decomposition theorem}
  \begin{proofs}[Proof sketch]
    Let \(U \colon L^2(\mu) \to L^2(\mu)\) be the Koopman operator \(U(\varphi) = \varphi \circ g\). Let \(L \colon L^2(\mu) \to L^2(\mu)\) be its adjoint.
    \[
      \begin{tikzcd}
        {L^2(\mathcal{F}_i)} & {L^2(\mathcal{F}_{i+1})}
        \arrow["U", bend left, from=1-1, to=1-2]
        \arrow["L", bend left, from=1-2, to=1-1].
      \end{tikzcd}
    \]
    Define
    \begin{equation*}
      s = \sum_{i = 1}^{\infty} L^i(\CE[\mu]{p,\mathcal{F}_i})\quad\text{ and }\quad
      q = \sum_{i = 0}^{\infty} L^i(\CE[\mu]{p,\mathcal{F}_i} - \CE[\mu]{p, \mathcal{F}_{i + 1}}).
    \end{equation*}
  \end{proofs}
\end{frame}

\begin{frame}[fragile]
  \frametitle{A decomposition theorem}
  \begin{proof}[Proof sketch (cont.)]
    \vspace{-1.5em}
    \begin{align*}
      q &= \sum_{i = 0}^{\infty} L^i(\CE[\mu]{p,\mathcal{F}_i} - \CE[\mu]{p, \mathcal{F}_{i + 1}}) \\
        &=
          \CE[\mu]{p,\mathcal{F}_0} + 
          \sum_{i = 1}^{\infty} L^i(\CE[\mu]{p,\mathcal{F}_i}) - 
          \sum_{i = 0}^{\infty} L^i(\CE[\mu]{p, \mathcal{F}_{i + 1}})\\
        &=
          p +
          \sum_{i = 1}^{\infty} L^i(\CE[\mu]{p,\mathcal{F}_i}) - 
          \sum_{i = 0}^{\infty} UL(L^i(\CE[\mu]{p, \mathcal{F}_{i + 1}}))\\
        &=
          p +
          \sum_{i = 1}^{\infty} L^i(\CE[\mu]{p,\mathcal{F}_i}) - 
          \sum_{i = 1}^{\infty} L^i(\CE[\mu]{p, \mathcal{F}_{i}}) \circ g\\
        &=
          p + s - s \circ g.
    \end{align*}
    Also, \(L^i(\CE[\mu]{p,\mathcal{F}_i} - \CE[\mu]{p, \mathcal{F}_{i + 1}}) \in L^2(\mathcal{F}_1)^{\bot}\). Therefore, \(\CE[\mu]{q,\mathcal{F}_1} = 0\).
  \end{proof}
\end{frame}

\begin{frame}
  \frametitle{A decomposition theorem}
  For invertible \(g\):
  \begin{theorem}[{\textcite[Theorem 4.13]{vianaStochasticDynamicsDeterministic1997}}]
    Suppose \(p \in L^2(\mu)\), \(p\), \(g\) and \(g^{-1}\) are \(\mathcal{F}\)-measurable and there is a smaller \(\sigma\)-algebra \(\mathcal{G} \subseteq \mathcal{F}\) for which \(\mathcal{G}_n = g^{-n}(\mathcal{G})\) satisfies
    \begin{equation*}
      \sum_{n = 0}^{\infty} \lVert \CE[\mu]{p, \mathcal{G}_n} \rVert_2 < \infty\quad \text{ and }
      \quad \sum_{n = 0}^{\infty} \norm{p - \CE[\mu]{p,\mathcal{G}_{-n}}}_2 < \infty.
    \end{equation*}
    Then a reverse martingale-coboundary decomposition \(p = q + s \circ g - s\) exists.
  \end{theorem}
\end{frame}

\begin{frame}[fragile]
  \frametitle{A decomposition theorem}
  \begin{proofs}[Proof sketch]
    Let \(U,L \colon L^2(\mu) \to L^2(\mu)\) be adjoint to each other as before:
    \[
      \begin{tikzcd}
        {L^2(\mathcal{G}_i)} & {L^2(\mathcal{G}_{i+1})}
        \arrow["U", bend left, from=1-1, to=1-2]
        \arrow["L", bend left, from=1-2, to=1-1].
      \end{tikzcd}
    \]
    Introduce \(\OP[\mu]{\phi,\mathcal{A}} = \phi - \CE[\mu]{\phi,\mathcal{A}}\), the orthogonal projection onto \(L^2(\mathcal{A})^{\bot}\), and define
    \begin{equation*}
      s^+ = \sum_{i = 1}^{\infty} L^i(\CE[\mu]{p,\mathcal{G}_i})\quad\text{ and }\quad
      q^+ = \sum_{i = 0}^{\infty} L^i(\CE[\mu]{p,\mathcal{G}_i} - \CE[\mu]{p, \mathcal{G}_{i + 1}}),
    \end{equation*}
    \begin{equation*}
      s^- = - \sum_{i = 0}^{\infty} U^i(\OP[\mu]{p,\mathcal{G}_{-i}})\quad\text{ and }\quad
      q^- = - \sum_{i = 0}^{\infty} U^i(\OP[\mu]{p, \mathcal{G}_{-i}} - \OP[\mu]{p,\mathcal{G}_{-i + 1}}).
    \end{equation*}
    Like before, the \(q^+\) and \(q^-\) are reverse martingale difference sequences because each term in the series is in \(L^2(\mathcal{G}_1)^{\bot}\).
  \end{proofs}
\end{frame}
\begin{frame}[fragile]
  \frametitle{A decomposition theorem}
  \begin{proof}[Proof sketch (cont.)]
    With the same calculation as before, we arrive at \(q^+ = p^+ + s^+ - s^+ \circ g\), and
    \begin{align*}
      q^- &=
            \OP[\mu]{p,\mathcal{G}_0} + 
            \sum_{i = 1}^{\infty} U^i(\OP[\mu]{p,\mathcal{G}_{-i + 1}}) - 
            \sum_{i = 0}^{\infty} U^i(\OP[\mu]{p, \mathcal{G}_{-i}})\\
          &=
            p^- +
            \sum_{i = 0}^{\infty} U(U^i(\OP[\mu]{p,\mathcal{G}_{-i}})) - 
            \sum_{i = 0}^{\infty} U^i(\OP[\mu]{p, \mathcal{G}_{-i}})\\
          &=
            p^- +
            \sum_{i = 0}^{\infty} U^i(\OP[\mu]{p,\mathcal{G}_{-i}}) \circ g - 
            \sum_{i = 0}^{\infty} U^i(\OP[\mu]{p, \mathcal{G}_{-i}})\\
          &=
            p^- + s^- - s^- \circ g.
    \end{align*}
    So \(p = q + s \circ g - s\) for \(q = q^+ + q^-\) and \(s = s^+ + s^-\).
  \end{proof}
\end{frame}

\begin{frame}
  \frametitle{A decomposition theorem}
  \begin{lemma}
    Suppose that for a \(\mathcal{F}\)-measurable \(p \in L^{2}(\mu)\), there exists a summable sequence \(c_n > 0\) satisfying
    \begin{equation*}
      \int p \cdot (\psi \circ g^n) \;d\mu \leq \lVert \psi \rVert_2c_n
    \end{equation*}
    for all \(\psi \in L^2(\mathcal{F})\). Then
    \begin{equation*}
      \sum_{n = 0}^{\infty} \lVert \CE{p, g^{-n}(\mathcal{F})} \rVert_2 < \infty.
    \end{equation*}
  \end{lemma}
  \begin{proofs}
    Since \(\CE{\cdot, \mathcal{A}}\) is the orthogonal projection onto \(L^2(\mathcal{A})\),
    \begin{align*}
      \CE{p, g^{-n}(\mathcal{F})}
      &= \sup \set[par = \Big]{\int p \cdot \xi \;d\mu \where \xi \in L^2(g^{-n}(\mathcal{F})),\;\norm{\xi}_2 = 1}\\
      &= \sup \set[par = \Big]{\int p \cdot (\psi \circ g^n) \;d\mu \where \psi \in L^2(\mathcal{F}),\;\norm{\psi}_2 = 1}
      \le \norm{\psi}_2 c_n.
    \end{align*}\vspace*{-1em}
  \end{proofs}
\end{frame}
